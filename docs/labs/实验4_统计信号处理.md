### 任务1：随机信号的统计特性分析

生成随机信号并计算其统计量（均值、方差、自相关函数）。

```python
import numpy as np
import matplotlib.pyplot as plt

# 参数设置
fs = 1000          # 采样率
N = 1024           # 信号长度
t = np.arange(N) / fs

# 生成高斯白噪声信号
np.random.seed(42)
x = np.random.randn(N)

# TODO: 计算自相关函数（无偏估计）
def autocorr(x):
    """计算归一化自相关函数"""
    N = len(x)
    x = x - np.mean(x)
    r = np.correlate(x, x, mode='full')
    r = r / (____ * np.var(x))  # 归一化
    return r[N-1:]               # 取单边

r_xx = autocorr(x)
```

### 任务3：有色噪声的生成与分析

利用线性滤波器将白噪声转换为有色噪声，分析其频谱特性。

```python
from scipy.signal import lfilter, butter

# TODO: 生成粉色噪声（1/f噪声）—— 通过低通滤波白噪声
b, a = butter(3, ____, btype='low')  # 3阶低通滤波器，截止频率0.3（归一化）
pink_noise = lfilter(b, a, np.random.randn(N))

# TODO: 生成蓝色噪声（f噪声）—— 通过高通滤波白噪声
b_h, a_h = butter(3, ____, btype='high')  # 截止频率0.3
blue_noise = lfilter(b_h, a_h, np.random.randn(N))

# 计算各噪声的PSD并对比
f_w, Pxx_white = welch(np.random.randn(N), fs=fs, nperseg=256)
f_w, Pxx_pink = welch(pink_noise, fs=fs, nperseg=256)
f_w, Pxx_blue = welch(blue_noise, fs=fs, nperseg=256)

# TODO: 绘制三种噪声的PSD对比图
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# 时域波形
t_short = t[:200]
axes[0].plot(t_short, np.random.randn(200), alpha=0.7, label='白噪声')

axes[0].plot(t_short, blue_noise[:200], alpha=0.7, label='蓝色噪声')
axes[0].set_title('时域波形'); axes[0].legend()

# 频域PSD
axes[1].plot(f_w, 10*np.log10(Pxx_white), label='白噪声')

axes[1].set_title('PSD对比'); axes[1].set_xlabel('频率 (Hz)')
axes[1].set_ylabel('功率/频率 (dB/Hz)'); axes[1].legend()
```
