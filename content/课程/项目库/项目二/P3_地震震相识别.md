# P3 地震震相识别

## 一、项目背景

地震是地球内部能量释放引起的地壳振动，是地球科学中最基本的研究对象之一。地震波在地球内部传播时，会因介质变化而产生不同类型的波（震相）：

- **P波（纵波）**：压缩波，速度快（约6-8 km/s），最先到达，引起地面前后振动
- **S波（横波）**：剪切波，速度慢（约3.5-4.5 km/s），次先到达，引起地面左右振动
- **面波**：沿地表传播，速度最慢，振幅最大，破坏性最强

**地震早期预警（EEW）** 系统利用P波和S波到达的时间差，在破坏性S波到达前发出警报。准确的P波和S波到时自动检测是地震学的核心任务。

**核心挑战**：在低信噪比环境下自动、准确地检测P波和S波的到达时间，并区分地震信号与噪声。

## 二、信号处理建模

### 处理流程

```
三分量地震波形 (BHZ/BHN/BHE)
    ↓
带通滤波 (1-20 Hz)
    ↓
STA/LTA 触发算法 → 初步检测
    ↓
P波精确拾取 (AIC / AR-AIC)
    ↓
S波拾取 (偏振分析 / STA-LTA比)
    ↓
CNN 深度学习分类 (地震 vs 噪声 / P波 vs S波)
```

### STA/LTA 算法

短期平均/长期平均（STA/LTA）是最经典的地震事件检测算法：

$$STA(t) = \frac{1}{N_{STA}} \sum_{i=t-N_{STA}+1}^{t} |x(i)|$$

$$LTA(t) = \frac{1}{N_{LTA}} \sum_{i=t-N_{LTA}+1}^{t} |x(i)|$$

$$R(t) = \frac{STA(t)}{LTA(t)}$$

当 $R(t)$ 超过阈值时，触发地震事件检测。

```python
def generate_synthetic_earthquake(t, p_arrival=25, s_arrival=32):
    """生成合成地震记录"""
    # 背景噪声
    noise = np.random.normal(0, 0.05, (len(t), 3))

    p_signal[p_mask, 2] = 2.0 * p_env  # Z分量
    p_signal[p_mask, 0] = 0.5 * p_env  # E分量
    p_signal[p_mask, 1] = 0.5 * p_env  # N分量

    # S波信号（低频，水平分量为主）
    s_signal = np.zeros((len(t), 3))
    s_mask = t >= s_arrival
    s_env = np.exp(-0.3 * (t[s_mask] - s_arrival)) * np.sin(2 * np.pi * 4 * (t[s_mask] - s_arrival))
    s_signal[s_mask, 0] = 3.0 * s_env  # E分量
    s_signal[s_mask, 1] = 3.0 * s_env  # N分量
    s_signal[s_mask, 2] = 1.0 * s_env  # Z分量

    return noise + p_signal + s_signal
```

```python
def bandpass_filter(data, lowcut, highcut, fs, order=4):
    """带通滤波器"""
    nyq = 0.5 * fs
    low = lowcut / nyq
    high = highcut / nyq
    b, a = butter(order, [low, high], btype='band')
    if data.ndim == 1:
        return filtfilt(b, a, data)
    else:
        return np.array([filtfilt(b, a, data[:, i]) for i in range(data.shape[1])]).T
```

### 4.3 STA/LTA 触发算法

```python
def sta_lta_trigger(signal, fs, sta_window=0.5, lta_window=10, threshold=3.0):
    """
    STA/LTA 触发算法
    sta_window: 短时窗口（秒）
    lta_window: 长时窗口（秒）
    threshold: 触发阈值
    """
    sta_len = int(sta_window * fs)
    lta_len = int(lta_window * fs)

    # 计算特征函数（能量）
    energy = signal ** 2

    # 计算 STA 和 LTA
    sta = np.zeros_like(energy)
    lta = np.zeros_like(energy)

    # 使用递推算法提高效率
    sta[0] = energy[0]
    lta[0] = energy[0]

    for i in range(1, len(energy)):
        if i < sta_len:
            sta[i] = np.mean(energy[:i+1])
        else:
            sta[i] = (sta[i-1] * (sta_len - 1) + energy[i]) / sta_len

        if i < lta_len:
            lta[i] = np.mean(energy[:i+1])
        else:
            lta[i] = (lta[i-1] * (lta_len - 1) + energy[i]) / lta_len

    # 避免除零
    lta = np.maximum(lta, 1e-10)

    # STA/LTA 比值
    ratio = sta / lta

    # 检测触发
    triggers = np.where(ratio > threshold)[0]

    return ratio, triggers, sta, lta

# 对 Z 分量应用 STA/LTA
ratio, triggers, sta, lta = sta_lta_trigger(waveform_filt[:, 2], fs,
```

### 4.4 P波精确拾取（AIC方法）

```python
def aic_picker(signal, trigger_idx, search_window=200):
    """
    AIC (Akaike Information Criterion) 方法精确拾取P波
    AIC(k) = k*log(var(x[1:k])) + (N-k-1)*log(var(x[k+1:N]))
    """
    start = max(0, trigger_idx - search_window)
    end = min(len(signal), trigger_idx + search_window)
    x = signal[start:end]
    N = len(x)

    aic = np.zeros(N)
    for k in range(1, N-1):
        var1 = np.var(x[:k]) + 1e-10
        var2 = np.var(x[k:]) + 1e-10
        aic[k] = k * np.log(var1) + (N - k - 1) * np.log(var2)

    # AIC 最小值对应P波到时
    aic_pick = np.argmin(aic) + start
    return aic_pick, aic + start

# AIC 精确拾取
if len(triggers) > 0:
    aic_pick, aic_values = aic_picker(waveform_filt[:, 2], triggers[0])
    print(f"AIC 精确拾取: {t[aic_pick]:.3f} 秒")
```

### 4.5 能量比特征（震相识别）

```python
def compute_energy_envelope(signal, window=100):
    """计算能量包络"""
    energy = signal ** 2
    envelope = np.convolve(energy, np.ones(window)/window, mode='same')
    return envelope

def detect_s_wave(p_pick, signal_z, signal_e, signal_n, fs, search_after=5):
    """
    基于能量比的S波检测
    S波通常在P波后几秒到达，水平分量能量增大
    """
    search_start = p_pick + int(1 * fs)  # P波后1秒开始搜索
    search_end = min(len(signal_z), p_pick + int(search_after * fs))

    # 计算水平分量与垂直分量的能量比
    window = int(0.5 * fs)
    e_ratio = np.zeros(search_end - search_start)

    for i in range(len(e_ratio)):
        idx = search_start + i
        if idx + window > len(signal_z):
            break
        h_energy = np.sum(signal_e[idx:idx+window]**2 + signal_n[idx:idx+window]**2)
        v_energy = np.sum(signal_z[idx:idx+window]**2) + 1e-10
        e_ratio[i] = h_energy / v_energy

    # S波对应能量比突增
    s_pick_idx = np.argmax(e_ratio) + search_start
    return s_pick_idx, e_ratio

# 检测S波
if len(triggers) > 0:
    s_pick, e_ratio = detect_s_wave(triggers[0],
                                     waveform_filt[:, 2],
                                     waveform_filt[:, 0],
                                     waveform_filt[:, 1],
                                     fs)
    print(f"S波检测: {t[s_pick]:.3f} 秒")
    print(f"P-S到时差: {t[s_pick] - t[triggers[0]]:.3f} 秒")
```

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader

class SeismicCNN(nn.Module):
    """地震波形分类 CNN"""
    def __init__(self, n_classes=3):  # 噪声/P波/S波
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv1d(3, 32, kernel_size=7, padding=3),
            nn.ReLU(),
            nn.MaxPool1d(2),
            nn.Conv1d(32, 64, kernel_size=5, padding=2),
            nn.ReLU(),
            nn.MaxPool1d(2),
            nn.Conv1d(64, 128, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool1d(2),
            nn.Conv1d(128, 256, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool1d(1),
        )
        self.classifier = nn.Sequential(
            nn.Linear(256, 64),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(64, n_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

class SeismicDataset(Dataset):
    """地震数据集"""

        self.labels = torch.LongTensor(labels)

    def __len__(self):
        return len(self.labels)

    def __getitem__(self, idx):
        return self.waveforms[idx], self.labels[idx]
```
