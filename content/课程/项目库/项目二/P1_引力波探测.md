# P1 引力波探测

## 一、项目背景

引力波（Gravitational Wave）是爱因斯坦广义相对论预言的时空涟漪，由加速运动的质量产生。2015年9月14日，LIGO（激光干涉引力波天文台）首次直接探测到引力波信号 **GW150914**——来自两个黑洞的并合，这一发现获得了2017年诺贝尔物理学奖。

LIGO探测器本质上是一个巨型迈克尔逊干涉仪（臂长4km），引力波经过时会引起两臂长度的微小差异变化（约10⁻¹⁸米量级，小于质子直径的千分之一）。从如此微弱的信号中提取引力波信息，需要极其精密的信号处理技术。

**核心挑战**：引力波信号淹没在巨大的噪声背景中，需要匹配滤波、白化、时频分析等技术提取信号。

### 引力波信号特征

双黑洞并合信号（chirp信号）包含三个阶段：
1. **旋进（Inspiral）**：频率和振幅逐渐增大
2. **并合（Merger）**：频率和振幅达到峰值
3. **铃宕（Ringdown）**：频率和振幅快速衰减

## 二、信号处理建模

### 处理流程

```
原始应变数据 strain(t)
    ↓
计算噪声功率谱密度 ASD(f)
    ↓
白化滤波 (Whitening)
    ↓
带通滤波 (35-350 Hz)
    ↓
匹配滤波 (Matched Filter) ← 模板波形库
    ↓
信噪比 SNR(t) 时间序列
    ↓
Q变换 → 时频图 (Time-Frequency)
```

### 4.2 计算振幅谱密度（ASD）

```python
def compute_asd(strain, fs, segment_duration=4):
    """计算振幅谱密度"""
    from scipy.signal import welch
    nperseg = int(segment_duration * fs)
    freqs, psd = welch(strain, fs=fs, nperseg=nperseg, noverlap=nperseg//2)
    asd = np.sqrt(psd)
    return freqs, asd

freqs_h1, asd_h1 = compute_asd(strain_h1, fs)
freqs_l1, asd_l1 = compute_asd(strain_l1, fs)

# 绘制 ASD
fig, ax = plt.subplots(figsize=(12, 6))
ax.loglog(freqs_h1, asd_h1, label='H1 (Hanford)', alpha=0.8)
ax.loglog(freqs_l1, asd_l1, label='L1 (Livingston)', alpha=0.8)
ax.set_xlabel('频率 (Hz)')
ax.set_ylabel('ASD (strain/√Hz)')
ax.set_title('LIGO 噪声振幅谱密度')
```

### 4.3 白化滤波

```python
def whiten(strain, dt, fft_len=None):
    """
    白化滤波：使噪声功率谱变为平坦
    原理：白化后的信号在频域中除以 sqrt(PSD)
    """
    if fft_len is None:
        fft_len = 4 * int(1.0 / dt)

    # 计算 PSD
    freqs, psd = welch(strain, fs=1.0/dt, nperseg=fft_len)

    # 插值到完整频率网格
    freqs_full = np.fft.rfftfreq(len(strain), d=dt)
    psd_interp = interp1d(freqs, psd, bounds_error=False, fill_value='extrapolate')
    psd_full = psd_interp(freqs_full)
    psd_full = np.maximum(psd_full, 1e-60)  # 避免除零

    # 白化
    hf = np.fft.rfft(strain)
    white_hf = hf / np.sqrt(psd_full * dt)
    white = np.fft.irfft(white_hf, n=len(strain))

    return white

# 白化
white_h1 = whiten(strain_h1, dt)
white_l1 = whiten(strain_l1, dt)
```

### 4.4 带通滤波

```python
from scipy.signal import butter, filtfilt

def bandpass_filter(data, low, high, fs, order=5):
    """带通滤波"""
    nyq = 0.5 * fs
    b, a = butter(order, [low/nyq, high/nyq], btype='band')
    return filtfilt(b, a, data)

# 带通滤波 35-350 Hz（引力波信号主要频段）
```

### 4.5 匹配滤波

```python
def matched_filter(strain, template, psd, dt, low_freq=35):
    """
    匹配滤波器
    SNR(t) = IFFT[Template*(f) * Data(f) / PSD(f)]
    """
    # 频域变换
    data_freq = np.fft.rfft(strain) * dt
    template_freq = np.fft.rfft(template) * dt
    freqs = np.fft.rfftfreq(len(strain), d=dt)

    # 插值 PSD
    psd_interp = interp1d(np.arange(len(psd)),
                          psd, bounds_error=False, fill_value='extrapolate')

    # 频率掩码
    freq_mask = freqs >= low_freq

    # 计算 SNR
    integrand = np.zeros_like(freqs, dtype=complex)
    integrand[freq_mask] = (template_freq[freq_mask].conj() *
                            data_freq[freq_mask] / psd_interp(freqs[freq_mask]))

    # IFFT 得到 SNR 时间序列
    snr = np.fft.irfft(integrand, n=len(strain)) / dt

    # 归一化
    norm = np.sqrt(np.sum(np.abs(template_freq[freq_mask])**2 /
                           psd_interp(freqs[freq_mask])) * 4 * dt)
    snr = np.abs(snr) / norm

    return snr
```

### 4.7 信噪比估计

```python
# 估算探测信噪比
def estimate_snr(data_strain, template, psd_func, dt, low_freq=35):
    """简化的 SNR 估计"""
    N = len(data_strain)
    freqs = np.fft.rfftfreq(N, d=dt)

    data_fft = np.fft.rfft(data_strain) * dt
    temp_fft = np.fft.rfft(template) * dt
    psd_vals = psd_func(freqs)

    mask = freqs >= low_freq

    # SNR² = 4 ∫ |h(f)|² / S_n(f) df
    snr_squared = 4 * np.sum(np.abs(temp_fft[mask])**2 / psd_vals[mask]) * (1.0 / (N * dt))
    snr = np.sqrt(snr_squared)

    return snr
```
