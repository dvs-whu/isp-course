# H1 音乐流派识别

## 一、项目背景

音乐信息检索（Music Information Retrieval, MIR）是信号处理与人工智能的交叉领域，旨在从音频信号中自动提取音乐的语义信息。音乐流派识别（Music Genre Classification）是 MIR 中最经典的任务之一。

**人类如何识别音乐流派？**
- 节奏模式（摇滚的强拍 vs 爵士的切分）
- 音色特征（电吉他的失真 vs 钢琴的纯净）
- 旋律与和声特点
- 乐器组合

**机器如何识别？** 通过分析音频的频谱特征：
- **梅尔频率倒谱系数（MFCC）**：模拟人耳的频率感知
- **色度特征（Chroma）**：反映和声/调性信息
- **节奏特征**：tempo、beat strength
- **频谱特征**：频谱质心、带宽、rolloff

## 二、信号处理建模

### 处理流程

```
原始音频 (PCM, 22050 Hz)
    ↓
STFT → 频谱图 (Spectrogram)
    ↓
梅尔滤波器组 → 梅尔频谱图 (Mel Spectrogram)
    ↓
DCT → MFCC 特征
    ↓
特征聚合 (均值/方差/差分)
    ↓
CNN/RNN 分类器 → 流派标签
```

### 关键公式

**MFCC 计算步骤**：

1. STFT：$X(m, k) = \sum_{n=0}^{N-1} x(n+mH) w(n) e^{-j2\pi kn/N}$

2. 梅尔滤波：$S(m) = \sum_{k=0}^{N/2} |X(m,k)|^2 H_m(k)$

3. 对数：$\log S(m)$

4. DCT：$c_n = \sum_{m=0}^{M-1} \log S(m) \cos\left[\frac{\pi n(m+0.5)}{M}\right]$

**梅尔频率转换**：$f_{mel} = 2595 \log_{10}\left(1 + \frac{f}{700}\right)$

```python
# 特征聚合（整首歌的统计量）
def aggregate_features(mfccs, mfcc_delta, mfcc_delta2):
    """将时间序列特征聚合为固定长度向量"""
    features = []
    for feat in [mfccs, mfcc_delta, mfcc_delta2]:
        features.append(np.mean(feat, axis=1))    # 均值
        features.append(np.std(feat, axis=1))     # 标准差
        features.append(np.max(feat, axis=1))     # 最大值
        features.append(np.min(feat, axis=1))     # 最小值
    return np.concatenate(features)

feature_vector = aggregate_features(mfccs, mfcc_delta, mfcc_delta2)
```

### 4.5 完整特征提取流水线

```python
def extract_genre_features(y, sr):
    """提取完整的流派识别特征集"""
    features = {}

    # 1. MFCC
    mfccs = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13, n_fft=2048, hop_length=512)
    mfcc_delta = librosa.feature.delta(mfccs)
    mfcc_delta2 = librosa.feature.delta(mfccs, order=2)
    features['mfcc_mean'] = np.mean(mfccs, axis=1)
    features['mfcc_std'] = np.std(mfccs, axis=1)
    features['mfcc_delta_mean'] = np.mean(mfcc_delta, axis=1)
    features['mfcc_delta2_mean'] = np.mean(mfcc_delta2, axis=1)

    # 2. 色度特征 (Chroma) - 反映和声/调性
    chroma = librosa.feature.chroma_stft(y=y, sr=sr, n_fft=2048, hop_length=512)
    features['chroma_mean'] = np.mean(chroma, axis=1)
    features['chroma_std'] = np.std(chroma, axis=1)

    # 3. 频谱特征
    spec_cent = librosa.feature.spectral_centroid(y=y, sr=sr)[0]
    spec_bw = librosa.feature.spectral_bandwidth(y=y, sr=sr)[0]
    spec_rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr)[0]
    spec_contrast = librosa.feature.spectral_contrast(y=y, sr=sr)
    features['spectral_centroid'] = [np.mean(spec_cent), np.std(spec_cent)]
    features['spectral_bandwidth'] = [np.mean(spec_bw), np.std(spec_bw)]
    features['spectral_rolloff'] = [np.mean(spec_rolloff), np.std(spec_rolloff)]
    features['spectral_contrast_mean'] = np.mean(spec_contrast, axis=1)

    # 4. 节奏特征
    tempo, _ = librosa.beat.beat_track(y=y, sr=sr)
    features['tempo'] = [tempo]

    # 5. 零交叉率
    zcr = librosa.feature.zero_crossing_rate(y)[0]
    features['zcr'] = [np.mean(zcr), np.std(zcr)]

    # 6. RMS 能量
    rms = librosa.feature.rms(y=y)[0]
    features['rms'] = [np.mean(rms), np.std(rms)]

    # 合并所有特征
    all_features = np.concatenate([np.atleast_1d(v) for v in features.values()])
    return all_features

# 测试
```

### 4.6 CNN 深度学习分类

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader

class GenreCNN(nn.Module):
    """基于梅尔频谱图的音乐流派分类 CNN"""
    def __init__(self, n_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            # 输入: (1, 128, T) 梅尔频谱图
            nn.Conv2d(1, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.MaxPool2d(2, 2),
            nn.Dropout2d(0.2),

            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.MaxPool2d(2, 2),
            nn.Dropout2d(0.2),

            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.MaxPool2d(2, 2),
            nn.Dropout2d(0.3),

            nn.Conv2d(128, 256, kernel_size=3, padding=1),
            nn.BatchNorm2d(256),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((4, 4)),
            nn.Dropout2d(0.3),
        )
        self.classifier = nn.Sequential(
            nn.Linear(256 * 4 * 4, 512),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(512, 128),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(128, n_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

class MusicGenreDataset(Dataset):
    """音乐流派数据集"""
    def __init__(self, file_list, labels, sr=22050, duration=30, n_mels=128):
        self.file_list = file_list
        self.labels = labels
        self.sr = sr
        self.duration = duration
        self.n_mels = n_mels
        self.hop_length = 512

    def __len__(self):
        return len(self.file_list)

    def __getitem__(self, idx):
        # 加载音频
        y, _ = librosa.load(self.file_list[idx], sr=self.sr, duration=self.duration)

        # 计算梅尔频谱图
        mel = librosa.feature.melspectrogram(y=y, sr=self.sr,
                                               n_mels=self.n_mels,
                                               hop_length=self.hop_length)
        mel_db = librosa.power_to_db(mel, ref=np.max)

        # 归一化
        mel_db = (mel_db - mel_db.mean()) / (mel_db.std() + 1e-8)

        # 转为 tensor
        mel_tensor = torch.FloatTensor(mel_db).unsqueeze(0)  # (1, n_mels, T)
        label = torch.LongTensor([self.labels[idx]])

        return mel_tensor, label

# 模型实例化
model = GenreCNN(n_classes=10)
print("模型结构:")
print(model)
print(f"\n参数量: {sum(p.numel() for p in model.parameters()):,}")
```

```python
# 训练配置
def train_model(model, train_loader, val_loader, epochs=50, lr=0.001):
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    model = model.to(device)
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr, weight_decay=1e-4)
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer, patience=5)

    train_losses, val_losses = [], []

    for epoch in range(epochs):
        # 训练
        model.train()
        train_loss = 0
        for batch_mel, batch_label in train_loader:
            batch_mel = batch_mel.to(device)
            batch_label = batch_label.squeeze().to(device)

            outputs = model(batch_mel)
            loss = criterion(outputs, batch_label)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            train_loss += loss.item()

        # 验证
        model.eval()
        val_loss = 0
        correct = 0
        total = 0
        with torch.no_grad():
            for batch_mel, batch_label in val_loader:
                batch_mel = batch_mel.to(device)
                batch_label = batch_label.squeeze().to(device)
                outputs = model(batch_mel)
                loss = criterion(outputs, batch_label)
                val_loss += loss.item()
                _, predicted = torch.max(outputs, 1)
                total += batch_label.size(0)
                correct += (predicted == batch_label).sum().item()

        train_losses.append(train_loss / len(train_loader))
        val_losses.append(val_loss / len(val_loader))
        accuracy = 100 * correct / total

        scheduler.step(val_losses[-1])

        if (epoch + 1) % 10 == 0:
            print(f'Epoch [{epoch+1}/{epochs}], '
                  f'Train Loss: {train_losses[-1]:.4f}, '
                  f'Val Loss: {val_losses[-1]:.4f}, '
                  f'Val Acc: {accuracy:.2f}%')

    return train_losses, val_losses
```
