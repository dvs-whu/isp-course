## 一、项目背景

情感识别（Speech Emotion Recognition, SER）是语音信号处理与人工智能交叉的前沿领域。人类语音中蕴含丰富的情感信息——语调的高低起伏、语速的快慢变化、音量的强弱波动，都传递着说话者的内心状态。

### 应用场景
- **心理健康监测**：通过语音分析辅助抑郁症、焦虑症的早期筛查
- **智能客服**：自动识别用户情绪，实现情感化应答
- **方言保护**：分析地方戏曲、民歌中的情感特征，辅助非物质文化遗产数字化
- **人机交互**：让智能设备理解用户情感状态，提供更自然的交互体验

### 为什么用信号处理方法？
语音情感信息主要体现在**韵律特征**（基频、语速、能量）和**频谱特征**（共振峰、MFCC）中。通过信号处理技术提取这些特征，再结合机器学习模型，可以实现对情感的自动分类。

## 二、信号处理建模

### 2.1 处理流程

```
原始语音
  │
  ▼
预加重滤波 (Pre-emphasis)  ── 增强高频分量，补偿发声损耗
  │
  ▼
分帧加窗 (Framing & Windowing)  ── 短时平稳假设，25ms帧长，10ms帧移
  │
  ▼
特征提取
  ├── MFCC (梅尔频率倒谱系数)  ── 模拟人耳听觉特性
  ├── 基频 F0  ── 反映语调变化
  ├── 短时能量  ── 反映音量强弱
  └── 过零率  ── 反映清浊音比例
  │
  ▼
特征序列建模 (LSTM / CNN + Attention)
  │
  ▼
情感分类输出 (8类情感)
```

### 2.2 关键数学原理

**预加重滤波器**：
$$H(z) = 1 - \alpha z^{-1}, \quad \alpha \in [0.95, 0.97]$$

**MFCC 计算步骤**：
1. 对每帧信号做 FFT 得到频谱 $X(k)$
2. 通过 Mel 滤波器组 $H_m(k)$ 得到 Mel 频谱
3. 取对数：$S(m) = \ln\left(\sum_{k=0}^{N-1} |X(k)|^2 H_m(k)\right)$
4. 做 DCT 得到 MFCC：$c(n) = \sum_{m=0}^{M-1} S(m) \cos\left(\frac{\pi n(m+0.5)}{M}\right)$

**Mel 频率与线性频率转换**：
$$f_{mel} = 2595 \cdot \log_{10}\left(1 + \frac{f}{700}\right)$$

### 4.1 数据预处理

```python
import numpy as np
import librosa
import librosa.display
import matplotlib.pyplot as plt
from matplotlib import font_manager

# 设置中文字体
font_path = '/usr/share/fonts/truetype/noto/NotoSansCJKsc-Regular.otf'
font_prop = font_manager.FontProperties(fname=font_path)
plt.rcParams['font.family'] = font_prop.get_name()
plt.rcParams['axes.unicode_minus'] = False

def load_and_preprocess(audio_path, sr=22050):
    """加载并预处理音频"""
    # 加载音频
    y, sr = librosa.load(audio_path, sr=sr)
    
    # 预加重
    y_pre = np.append(y[0], y[1:] - 0.97 * y[:-1])
    
    # 归一化
    y_pre = y_pre / np.max(np.abs(y_pre))
    
    return y_pre, sr
```

### 4.2 MFCC 特征提取

```python
def extract_features(y, sr, n_mfcc=40):
    """提取完整的语音情感特征集"""
    features = {}
    
    # MFCC 特征
    mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc)
    mfcc_delta = librosa.feature.delta(mfcc)      # 一阶差分
    mfcc_delta2 = librosa.feature.delta(mfcc, order=2)  # 二阶差分
    
    # 统计特征：均值和标准差
    features['mfcc_mean'] = np.mean(mfcc, axis=1)
    features['mfcc_std'] = np.std(mfcc, axis=1)
    features['mfcc_delta_mean'] = np.mean(mfcc_delta, axis=1)
    features['mfcc_delta2_mean'] = np.mean(mfcc_delta2, axis=1)
    
    # 基频 F0
    f0, voiced_flag, voiced_probs = librosa.pyin(
        y, fmin=librosa.note_to_hz('C2'), fmax=librosa.note_to_hz('C7')
    )
    f0_clean = f0[~np.isnan(f0)]
    features['f0_mean'] = np.mean(f0_clean) if len(f0_clean) > 0 else 0
    features['f0_std'] = np.std(f0_clean) if len(f0_clean) > 0 else 0
    features['f0_range'] = np.ptp(f0_clean) if len(f0_clean) > 0 else 0
    
    # 短时能量
    energy = librosa.feature.rms(y=y)
    features['energy_mean'] = np.mean(energy)
    features['energy_std'] = np.std(energy)
    
    # 过零率
    zcr = librosa.feature.zero_crossing_rate(y)
    features['zcr_mean'] = np.mean(zcr)
    
    # 频谱质心和带宽
    spectral_centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
    features['spectral_centroid_mean'] = np.mean(spectral_centroid)
    
    return features
```

### 4.4 LSTM 情感分类模型

```python
import torch
import torch.nn as nn

class EmotionLSTM(nn.Module):
    """基于 LSTM 的语音情感识别模型"""
    def __init__(self, input_dim=40, hidden_dim=128, num_layers=2, num_classes=8):
        super(EmotionLSTM, self).__init__()
        self.lstm = nn.LSTM(
            input_dim, hidden_dim, num_layers,
            batch_first=True, dropout=0.3, bidirectional=True
        )
        self.attention = nn.Sequential(
            nn.Linear(hidden_dim * 2, 64),
            nn.Tanh(),
            nn.Linear(64, 1)
        )
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim * 2, 64),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(64, num_classes)
        )
    
    def forward(self, x):
        # x: (batch, seq_len, input_dim)
        lstm_out, _ = self.lstm(x)  # (batch, seq_len, hidden*2)
        
        # Attention 机制
        attn_weights = self.attention(lstm_out)  # (batch, seq_len, 1)
        attn_weights = torch.softmax(attn_weights, dim=1)
        context = torch.sum(lstm_out * attn_weights, dim=1)  # (batch, hidden*2)
        
        output = self.classifier(context)
        return output
```

### 4.5 完整训练流程

```python
from torch.utils.data import Dataset, DataLoader
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import LabelEncoder
import os

class RAVDESSDataset(Dataset):
    """RAVDESS 数据集加载器"""
    def __init__(self, root_dir, sr=22050, max_len=300):
        self.files = []
        self.labels = []
        self.sr = sr
        self.max_len = max_len
        
        # RAVDESS 文件名编码: 03-01-01-01-01-01-01.wav
        # 第三位为情感代码：01=neutral, 02=calm, 03=happy, 04=sad,
        #                     05=angry, 06=fearful, 07=disgust, 08=surprised
        for actor_dir in os.listdir(root_dir):
            actor_path = os.path.join(root_dir, actor_dir)
            if not os.path.isdir(actor_path):
                continue
            for fname in os.listdir(actor_path):
                if fname.endswith('.wav'):
                    emotion_id = int(fname.split('-')[2]) - 1  # 0-indexed
                    self.files.append(os.path.join(actor_path, fname))
                    self.labels.append(emotion_id)
        
        self.label_names = ['neutral', 'calm', 'happy', 'sad',
                           'angry', 'fearful', 'disgust', 'surprised']
    
    def __len__(self):
        return len(self.files)
    
    def __getitem__(self, idx):
        y, sr = librosa.load(self.files[idx], sr=self.sr)
        
        # 提取 MFCC 序列
        mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=40)
        mfcc = mfcc.T  # (time, n_mfcc)
        
        # 填充/截断到固定长度
        if mfcc.shape[0] < self.max_len:
            pad = np.zeros((self.max_len - mfcc.shape[0], 40))
            mfcc = np.vstack([mfcc, pad])
        else:
            mfcc = mfcc[:self.max_len]
        
        return torch.FloatTensor(mfcc), self.labels[idx]

# 训练
def train_emotion_model(root_dir, epochs=50, batch_size=32, lr=0.001):
    dataset = RAVDESSDataset(root_dir)
    train_set, val_set = train_test_split(dataset, test_size=0.2, 
                                          stratify=dataset.labels, random_state=42)
    
    train_loader = DataLoader(train_set, batch_size=batch_size, shuffle=True)
    val_loader = DataLoader(val_set, batch_size=batch_size)
    
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    model = EmotionLSTM(num_classes=8).to(device)
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer, patience=5)
    
    for epoch in range(epochs):
        model.train()
        total_loss, correct, total = 0, 0, 0
        
        for features, labels in train_loader:
            features, labels = features.to(device), labels.to(device)
            
            optimizer.zero_grad()
            outputs = model(features)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
            _, predicted = torch.max(outputs, 1)
            correct += (predicted == labels).sum().item()
            total += labels.size(0)
        
        train_acc = correct / total
        
        # 验证
        model.eval()
        val_correct, val_total = 0, 0
        with torch.no_grad():
            for features, labels in val_loader:
                features, labels = features.to(device), labels.to(device)
                outputs = model(features)
                _, predicted = torch.max(outputs, 1)
                val_correct += (predicted == labels).sum().item()
                val_total += labels.size(0)
        
        val_acc = val_correct / val_total
        scheduler.step(total_loss)
        
        print(f'Epoch [{epoch+1}/{epochs}]  '
              f'Loss: {total_loss/len(train_loader):.4f}  '
              f'Train Acc: {train_acc:.4f}  Val Acc: {val_acc:.4f}')
    
    return model
```
