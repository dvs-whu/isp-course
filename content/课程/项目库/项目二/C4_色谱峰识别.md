# C4 色谱峰识别

## 一、项目背景

色谱法（Chromatography）是分析化学中最重要的分离分析技术之一，广泛应用于药物分析、环境监测、食品安全、石油化工等领域。气相色谱（GC）和液相色谱（HPLC）通过将混合物在固定相和流动相之间的分配差异实现分离，输出色谱图（chromatogram）——检测器响应随时间变化的信号。

色谱峰的自动识别和定量分析是色谱数据处理的核心任务：
- **定性分析**：通过保留时间识别化合物
- **定量分析**：通过峰面积/峰高计算浓度

**核心挑战**：实际色谱信号存在基线漂移、噪声干扰、峰重叠（共流出）、拖尾/前伸等问题，需要信号处理技术进行精确的峰检测和积分。

## 二、信号处理建模

### 处理流程

```
原始色谱信号 (检测器响应 vs 时间)
    ↓
预处理 (平滑去噪)
    ↓
基线校正 (多项式拟合 / 形态学方法)
    ↓
峰检测 (一阶/二阶导数法、阈值法)
    ↓
峰拟合 (高斯/EMG模型)
    ↓
峰面积积分 → 浓度计算
```

### 数学模型

**高斯峰模型**：
$$h(t) = A \cdot \exp\left(-\frac{(t-t_R)^2}{2\sigma^2}\right)$$

**指数修正高斯（EMG）模型**（考虑拖尾）：
$$h(t) = \frac{A}{2\tau} \exp\left(\frac{\sigma^2}{2\tau^2} - \frac{t-t_R}{\tau}\right) \text{erfc}\left(\frac{1}{\sqrt{2}}\left(\frac{\sigma}{\tau} - \frac{t-t_R}{\sigma}\right)\right)$$

## 四、数据处理方法

### 4.1 模拟色谱数据生成

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.signal import find_peaks, savgol_filter
from scipy.optimize import curve_fit
from scipy.integrate import trapezoid

# 设置中文字体
plt.rcParams['font.family'] = 'Noto Sans CJK SC'
plt.rcParams['font.sans-serif'] = ['Noto Sans CJK SC']
plt.rcParams['axes.unicode_minus'] = False

def gaussian_peak(t, amplitude, t_R, sigma):
    """高斯峰模型"""
    return amplitude * np.exp(-(t - t_R)**2 / (2 * sigma**2))

def emg_peak(t, amplitude, t_R, sigma, tau):
    """指数修正高斯峰模型（拖尾峰）"""
    from scipy.special import erfc
    z = (sigma / tau - (t - t_R) / sigma) / np.sqrt(2)
    return (amplitude / (2 * tau)) * np.exp(sigma**2 / (2*tau**2) - (t-t_R)/tau) * erfc(z)

# 生成模拟色谱数据
np.random.seed(42)
t = np.linspace(0, 60, 3000)  # 0-60分钟，3000个点

# 定义色谱峰 (保留时间, 幅度, 半峰宽)
peaks_params = [
    (5.2,  0.8, 0.3),   # 峰1
    (12.1, 1.5, 0.4),   # 峰2
    (18.5, 0.6, 0.25),  # 峰3
    (25.3, 2.0, 0.5),   # 峰4
    (35.7, 1.2, 0.35),  # 峰5
    (42.0, 0.9, 0.3),   # 峰6
]

# 合成信号
signal = np.zeros_like(t)
for t_R, amp, sigma in peaks_params:
    signal += gaussian_peak(t, amp, t_R, sigma)

# 添加基线漂移
baseline = 0.05 * np.sin(2 * np.pi * t / 80) + 0.02 * t / 60
signal_with_baseline = signal + baseline

# 添加噪声
noise = np.random.normal(0, 0.03, len(t))
chromatogram = signal_with_baseline + noise
```

### 4.2 平滑与基线校正

```python
# 1. Savitzky-Golay 平滑
smoothed = savgol_filter(chromatogram, window_length=21, polyorder=3)

# 2. 基线校正 - 非对称最小二乘法（ALS）简化版
def baseline_als(y, lam=1e6, p=0.01, n_iter=10):
    """非对称最小二乘基线校正"""
    from scipy import sparse
    from scipy.sparse.linalg import spsolve
    L = len(y)
    D = sparse.diags([1, -2, 1], [0, -1, -2], shape=(L, L-2))
    D = lam * D.dot(D.T)
    w = np.ones(L)
    for i in range(n_iter):
        W = sparse.spdiags(w, 0, L, L)
        Z = W + D
        z = spsolve(Z, w * y)
        w = p * (y > z) + (1-p) * (y < z)
    return z

baseline_est = baseline_als(smoothed)
corrected = smoothed - baseline_est

# 可视化
fig, axes = plt.subplots(3, 1, figsize=(14, 10))

axes[0].plot(t, chromatogram, 'b-', alpha=0.7, label='原始色谱')
axes[0].set_title('原始色谱信号（含噪声和基线漂移）')
axes[0].set_ylabel('检测器响应')
axes[0].legend()

axes[1].plot(t, smoothed, 'g-', label='SG平滑后')
axes[1].plot(t, baseline_est, 'r--', linewidth=2, label='估计基线')
axes[1].set_title('平滑与基线估计')
axes[1].set_ylabel('响应')
axes[1].legend()

axes[2].plot(t, corrected, 'b-', label='基线校正后')
axes[2].set_title('基线校正后的色谱')
axes[2].set_xlabel('保留时间 (min)')
axes[2].set_ylabel('响应')
axes[2].legend()

plt.tight_layout()
plt.savefig('chromatogram_preprocessing.png', dpi=150, bbox_inches='tight')
plt.show()
```

### 4.3 峰检测

```python
# 使用 scipy find_peaks 进行峰检测
peaks, properties = find_peaks(corrected,
                                height=0.1,        # 最小峰高
                                prominence=0.05,   # 最小突出度
                                width=5,           # 最小宽度（采样点）
                                distance=50)       # 最小峰间距（采样点）

print(f"检测到 {len(peaks)} 个色谱峰")
print(f"峰位置（保留时间）: {t[peaks].round(2)} min")
print(f"峰高度: {corrected[peaks].round(4)}")

# 可视化检测结果
fig, ax = plt.subplots(figsize=(14, 5))
ax.plot(t, corrected, 'b-', label='色谱信号')
ax.plot(t[peaks], corrected[peaks], 'rv', markersize=10, label='检测到的峰')
ax.set_xlabel('保留时间 (min)')
ax.set_ylabel('响应')
ax.set_title('色谱峰检测结果')
ax.legend()

# 标注每个峰
for i, peak_idx in enumerate(peaks):
    ax.annotate(f'峰{i+1}\nt={t[peak_idx]:.1f}min',
                xy=(t[peak_idx], corrected[peak_idx]),
                xytext=(0, 20), textcoords='offset points',
                ha='center', fontsize=8,
                arrowprops=dict(arrowstyle='->', color='gray'))

plt.savefig('peak_detection.png', dpi=150, bbox_inches='tight')
plt.show()
```

### 4.4 峰拟合与面积积分

```python
def fit_and_integrate(t, signal, peak_idx, window=100):
    """对单个峰进行高斯拟合并计算面积"""
    # 提取峰周围的数据
    start = max(0, peak_idx - window)
    end = min(len(t), peak_idx + window)
    t_segment = t[start:end]
    s_segment = signal[start:end]

    # 初始参数估计
    amp_init = s_segment.max()
    tR_init = t[peak_idx]
    sigma_init = 0.5

    try:
        popt, pcov = curve_fit(gaussian_peak, t_segment, s_segment,
                               p0=[amp_init, tR_init, sigma_init],
                               maxfev=5000)
        t_fit = np.linspace(t_segment[0], t_segment[-1], 1000)
        s_fit = gaussian_peak(t_fit, *popt)

        # 面积积分（解析解：A * sigma * sqrt(2*pi)）
        area_analytical = popt[0] * popt[2] * np.sqrt(2 * np.pi)
        # 数值积分
        area_numerical = trapezoid(s_fit, t_fit)

        return {
            'amplitude': popt[0],
            'retention_time': popt[1],
            'sigma': popt[2],
            'area': area_analytical,
            't_fit': t_fit,
            's_fit': s_fit,
            'pcov': pcov
        }
    except RuntimeError:
        print(f"峰 {peak_idx} 拟合失败")
        return None

# 对每个峰进行拟合
results = []
for i, peak_idx in enumerate(peaks):
    result = fit_and_integrate(t, corrected, peak_idx)
    if result:
        results.append(result)
        print(f"峰{i+1}: 保留时间={result['retention_time']:.2f}min, "
              f"面积={result['area']:.4f}, σ={result['sigma']:.3f}")

# 可视化拟合结果
fig, ax = plt.subplots(figsize=(14, 5))
ax.plot(t, corrected, 'b-', alpha=0.5, label='色谱信号')
for i, res in enumerate(results):
    ax.plot(res['t_fit'], res['s_fit'], 'r-', linewidth=2, alpha=0.8)
    ax.fill_between(res['t_fit'], res['s_fit'], alpha=0.2, label=f'峰{i+1}')
ax.set_xlabel('保留时间 (min)')
ax.set_ylabel('响应')
ax.set_title('色谱峰拟合与面积积分')
ax.legend()
plt.savefig('peak_fitting.png', dpi=150, bbox_inches='tight')
plt.show()
```

### 4.5 定量分析（标准曲线法）

```python
# 标准曲线法示例
def calibration_analysis(peak_areas, concentrations):
    """建立标准曲线并计算未知样品浓度"""
    # 线性拟合
    coeffs = np.polyfit(concentrations, peak_areas, 1)
    slope, intercept = coeffs

    # 计算 R²
    y_pred = slope * concentrations + intercept
    ss_res = np.sum((peak_areas - y_pred)**2)
    ss_tot = np.sum((peak_areas - np.mean(peak_areas))**2)
    r_squared = 1 - ss_res / ss_tot

    print(f"标准曲线: y = {slope:.4f}x + {intercept:.4f}")
    print(f"R² = {r_squared:.6f}")

    return slope, intercept, r_squared
```

## 五、涉及知识点

| 知识点 | 说明 | 在本项目中的应用 |
|--------|------|------------------|
| **卷积与反卷积** | 色谱峰形是仪器响应函数与真实信号的卷积 | 峰反卷积解决重叠峰问题 |
| **滤波** | Savitzky-Golay 滤波器 | 色谱信号平滑去噪，保留峰形 |
| **曲线拟合** | 非线性最小二乘拟合 | 高斯/EMG 模型拟合色谱峰 |
| 导数光谱法 | 信号的导数分析 | 利用一阶/二阶导数检测峰位置 |
| 数值积分 | 梯形法、辛普森法 | 计算色谱峰面积用于定量 |
