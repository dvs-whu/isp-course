# E3 鸟类声音监测

> **项目类型**：环境交叉项目 | **难度**：⭐⭐⭐⭐ | **关键词**：生物声学、时频分析、声音事件检测

---

## 一、项目背景

鸟类是生态系统健康的重要指示物种。传统的鸟类调查依赖人工观察，耗时费力且覆盖范围有限。**被动声学监测**（Passive Acoustic Monitoring, PAM）通过自动录音设备持续采集环境声音，利用信号处理和机器学习技术自动识别鸟类叫声，为生物多样性监测提供了革命性的工具。

### 核心挑战
- **噪声干扰**：野外录音包含风声、虫鸣、水流等背景噪声
- **多物种混叠**：多种鸟类可能同时鸣叫
- **时间稀疏性**：鸟类叫声在长时间录音中占比很低
- **种内变异**：同一物种的叫声也存在地域和个体差异

### 信号处理视角
鸟类声音信号的分析需要**时频联合分析**：
- 时域：检测鸣叫发生的**时间位置**
- 频域：识别不同物种的**频率特征**（频率范围、音调模式）
- 时频域：通过频谱图捕捉**频率随时间变化**的模式

## 二、信号处理建模

### 2.1 处理流程

```
长时间环境录音 x(n)
  │
  ▼
预处理
  ├── 带通滤波（去除超低频和超高频噪声）
  ├── 自适应降噪
  └── 音量归一化
  │
  ▼
时频分析
  ├── STFT → 短时频谱图 S(t, f)
  ├── Mel 频谱图（模拟听觉特性）
  └── 常数Q变换 CQT（音乐性频谱）
  │
  ▼
声音事件检测
  ├── 能量阈值法
  ├── 频谱熵检测
  └── 基于CNN的检测器
  │
  ▼
物种分类
  ├── 频谱图 → CNN 特征提取 → Softmax 分类
  └── 音频片段 → 预训练模型（BirdNET）→ 物种识别
  │
  ▼
物种多样性报告 (种类、数量、时间分布)
```

### 2.2 关键数学原理

**短时傅里叶变换（STFT）**：
$$S(m, k) = \sum_{n=0}^{N-1} x(n + mH) \cdot w(n) \cdot e^{-j2\pi kn/N}$$
其中 $H$ 为帧移，$w(n)$ 为窗函数

**Mel 频谱**：
$$Mel(f) = 2595 \cdot \log_{10}(1 + f/700)$$

**频谱熵**（声音事件检测）：
$$H(m) = -\sum_{k} p(m,k) \log p(m,k)$$
其中 $p(m,k) = |S(m,k)|^2 / \sum_k |S(m,k)|^2$

### 4.1 音频加载与预处理

```python
import numpy as np
import librosa
import librosa.display
import soundfile as sf
import matplotlib.pyplot as plt
from matplotlib import font_manager
from scipy import signal

# 设置中文字体
font_path = '/usr/share/fonts/truetype/noto/NotoSansCJKsc-Regular.otf'
font_prop = font_manager.FontProperties(fname=font_path)
plt.rcParams['font.family'] = font_prop.get_name()
plt.rcParams['axes.unicode_minus'] = False

def load_audio(audio_path, sr=32000):
    """加载鸟类声音录音"""
    y, sr = librosa.load(audio_path, sr=sr)
    return y, sr

def bandpass_filter(y, sr, low_freq=500, high_freq=10000, order=5):
    """
    带通滤波：保留鸟类典型频率范围
    
    大多数鸟类鸣叫频率范围：1kHz - 10kHz
    """
    nyq = sr / 2
    low = low_freq / nyq
    high = high_freq / nyq
    
    b, a = signal.butter(order, [low, high], btype='band')
    y_filtered = signal.filtfilt(b, a, y)
    
    return y_filtered

def normalize_audio(y):
    """归一化到 [-1, 1]"""
    return y / np.max(np.abs(y))
```

### 4.2 时频分析 — 生成频谱图

```python
def compute_spectrograms(y, sr):
    """计算多种频谱图表示"""
    spectrograms = {}
    
    # 1. STFT 频谱图
    S = np.abs(librosa.stft(y, n_fft=2048, hop_length=512))
    S_db = librosa.amplitude_to_db(S, ref=np.max)
    spectrograms['stft'] = S_db
    
    # 2. Mel 频谱图
    mel_spec = librosa.feature.melspectrogram(y=y, sr=sr, n_mels=128, fmax=sr//2)
    mel_db = librosa.power_to_db(mel_spec, ref=np.max)
    spectrograms['mel'] = mel_db
    
    # 3. CQT 频谱图（常数Q变换，频率分辨率随频率变化）
    cqt = np.abs(librosa.cqt(y, sr=sr, hop_length=512))
    cqt_db = librosa.amplitude_to_db(cqt, ref=np.max)
    spectrograms['cqt'] = cqt_db
    
    return spectrograms

def visualize_spectrograms(y, sr, title="鸟类声音"):
    """可视化不同类型的频谱图"""
    specs = compute_spectrograms(y, sr)
    
    fig, axes = plt.subplots(2, 2, figsize=(14, 8))
    
    # 波形
    times = np.arange(len(y)) / sr
    axes[0, 0].plot(times, y, color='steelblue', linewidth=0.3)
    axes[0, 0].set_title('波形', fontproperties=font_prop)
    axes[0, 0].set_xlabel('时间 (s)', fontproperties=font_prop)
    axes[0, 0].set_ylabel('幅度', fontproperties=font_prop)
    
    # STFT
    img1 = librosa.display.specshow(specs['stft'], sr=sr, hop_length=512,
                                     x_axis='time', y_axis='hz', ax=axes[0, 1])
    axes[0, 1].set_title('STFT 频谱图', fontproperties=font_prop)
    axes[0, 1].set_ylim(0, 12000)
    fig.colorbar(img1, ax=axes[0, 1], format='%+2.0f dB')
    
    # Mel
    img2 = librosa.display.specshow(specs['mel'], sr=sr, hop_length=512,
                                     x_axis='time', y_axis='mel', ax=axes[1, 0])
    axes[1, 0].set_title('Mel 频谱图', fontproperties=font_prop)
    fig.colorbar(img2, ax=axes[1, 0], format='%+2.0f dB')
    
    # CQT
    img3 = librosa.display.specshow(specs['cqt'], sr=sr, hop_length=512,
                                     x_axis='time', y_axis='cqt_hz', ax=axes[1, 1])
    axes[1, 1].set_title('CQT 频谱图', fontproperties=font_prop)
    fig.colorbar(img3, ax=axes[1, 1], format='%+2.0f dB')
    
    plt.suptitle(title, fontproperties=font_prop, fontsize=14)
    plt.tight_layout()
    plt.savefig(f'spectrograms_{title}.png', dpi=150, bbox_inches='tight')
    plt.show()
```

### 4.3 声音事件检测

```python
def detect_bird_calls_energy(y, sr, frame_length=2048, hop_length=512,
                              threshold_db=-40, min_duration=0.1):
    """
    基于能量的声音事件检测
    
    原理：鸟类鸣叫段的能量显著高于背景噪声
    """
    # 计算短时能量
    rms = librosa.feature.rms(y=y, frame_length=frame_length, hop_length=hop_length)[0]
    rms_db = 20 * np.log10(rms + 1e-10)
    
    # 自适应阈值
    noise_floor = np.percentile(rms_db, 20)  # 底部 20% 作为噪声估计
    threshold = noise_floor + abs(threshold_db - noise_floor) * 0.5
    
    # 检测超过阈值的帧
    is_active = rms_db > threshold
    
    # 合并相邻帧
    min_frames = int(min_duration * sr / hop_length)
    events = []
    start = None
    
    for i in range(len(is_active)):
        if is_active[i] and start is None:
            start = i
        elif not is_active[i] and start is not None:
            if (i - start) >= min_frames:
                start_time = start * hop_length / sr
                end_time = i * hop_length / sr
                events.append((start_time, end_time))
            start = None
    
    return events, rms_db, threshold

def detect_bird_calls_spectral_entropy(y, sr, frame_length=2048, hop_length=512,
                                        threshold=None):
    """
    基于频谱熵的声音事件检测
    
    原理：鸟类鸣叫的频谱更"集中"（低熵），而噪声更"均匀"（高熵）
    """
    S = np.abs(librosa.stft(y, n_fft=frame_length, hop_length=hop_length))
    
    # 计算每帧的频谱熵
    entropy = np.zeros(S.shape[1])
    for i in range(S.shape[1]):
        spectrum = S[:, i]
        spectrum = spectrum / (np.sum(spectrum) + 1e-10)
        entropy[i] = -np.sum(spectrum * np.log2(spectrum + 1e-10))
    
    # 自适应阈值
    if threshold is None:
        threshold = np.percentile(entropy, 30)  # 低熵 = 鸟叫
    
    is_call = entropy < threshold
    
    return is_call, entropy, threshold

def visualize_detection(y, sr, events):
    """可视化检测结果"""
    fig, axes = plt.subplots(3, 1, figsize=(14, 8))
    
    # 波形 + 检测框
    times = np.arange(len(y)) / sr
    axes[0].plot(times, y, color='steelblue', linewidth=0.3)
    for start, end in events:
        axes[0].axvspan(start, end, alpha=0.3, color='red')
    axes[0].set_title('波形与检测结果', fontproperties=font_prop)
    axes[0].set_xlabel('时间 (s)', fontproperties=font_prop)
    
    # 频谱图 + 检测框
    S = np.abs(librosa.stft(y, n_fft=2048, hop_length=512))
    S_db = librosa.amplitude_to_db(S, ref=np.max)
    img = librosa.display.specshow(S_db, sr=sr, hop_length=512,
                                    x_axis='time', y_axis='hz', ax=axes[1])
    for start, end in events:
        axes[1].axvspan(start, end, alpha=0.3, color='red')
    axes[1].set_ylim(0, 12000)
    axes[1].set_title('频谱图与检测框', fontproperties=font_prop)
    fig.colorbar(img, ax=axes[1], format='%+2.0f dB')
    
    # 能量曲线
    rms = librosa.feature.rms(y=y, frame_length=2048, hop_length=512)[0]
    rms_times = librosa.frames_to_time(np.arange(len(rms)), sr=sr, hop_length=512)
    axes[2].plot(rms_times, 20*np.log10(rms + 1e-10), 'b-', linewidth=1)
    axes[2].set_title('短时能量', fontproperties=font_prop)
    axes[2].set_xlabel('时间 (s)', fontproperties=font_prop)
    axes[2].set_ylabel('能量 (dB)', fontproperties=font_prop)
    
    plt.tight_layout()
    plt.savefig('bird_detection.png', dpi=150, bbox_inches='tight')
    plt.show()
```

### 4.4 CNN 物种分类模型

```python
import torch
import torch.nn as nn
import torchvision.transforms as transforms

class BirdSpeciesCNN(nn.Module):
    """基于 Mel 频谱图的鸟类物种分类 CNN"""
    def __init__(self, num_classes=10):
        super(BirdSpeciesCNN, self).__init__()
        
        self.features = nn.Sequential(
            # Conv Block 1
            nn.Conv2d(1, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.Conv2d(32, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Dropout2d(0.2),
            
            # Conv Block 2
            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.Conv2d(64, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Dropout2d(0.2),
            
            # Conv Block 3
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.Conv2d(128, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((4, 4)),
        )
        
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128 * 4 * 4, 256),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(256, num_classes)
        )
    
    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return x

def prepare_mel_spectrogram_dataset(audio_dir, labels, sr=32000, 
                                      duration=5, n_mels=128):
    """将音频文件转换为 Mel 频谱图数据集"""
    specs = []
    
    for audio_path in audio_dir:
        y, _ = librosa.load(audio_path, sr=sr, duration=duration)
        
        # 固定长度
        target_len = sr * duration
        if len(y) < target_len:
            y = np.pad(y, (0, target_len - len(y)))
        else:
            y = y[:target_len]
        
        # Mel 频谱图
        mel = librosa.feature.melspectrogram(y=y, sr=sr, n_mels=n_mels)
        mel_db = librosa.power_to_db(mel, ref=np.max)
        
        specs.append(mel_db)
    
    specs = np.array(specs)
    specs = (specs - specs.mean()) / (specs.std() + 1e-10)
    
    return torch.FloatTensor(specs).unsqueeze(1)  # (N, 1, n_mels, time)
```

### 4.5 使用 BirdNET 预训练模型

```python
def use_birdnet(audio_path):
    """
    使用 BirdNET 预训练模型进行鸟类识别
    
    BirdNET 是一个专门训练的鸟类声音识别深度学习模型
    支持 6000+ 物种
    """
    # 安装: pip install birdnetlib
    try:
        from birdnetlib.analyzer import Analyzer
        from birdnetlib import Recording
        
        analyzer = Analyzer()
        recording = Recording(
            analyzer,
            audio_path,
            min_confidence=0.3
        )
        recording.analyze()
        
        for detection in recording.detections:
            print(f"物种: {detection['common_name']} ({detection['scientific_name']})")
            print(f"  置信度: {detection['confidence']:.3f}")
            print(f"  时间: {detection['start_time']:.1f}s - {detection['end_time']:.1f}s")
        
        return recording.detections
    except ImportError:
        print("请安装 birdnetlib: pip install birdnetlib")
        return []
```
