# P4 声学共振频率

## 一、项目背景

每种材料都有独特的声学特性，当受到敲击激励时会产生特定的振动模式和共振频率。这种现象在日常生活中无处不在——敲击玻璃杯、金属板、木头会听到不同的声音。

**共振理论**：物体的固有频率由其形状、尺寸、密度和弹性模量决定。对于简单几何体，固有频率可以解析求解：

- 弦的振动：$f_n = \frac{n}{2L}\sqrt{\frac{T}{\mu}}$
- 板的振动：$f_{mn} = \frac{\pi}{2}\sqrt{\frac{D}{\rho h}}\left[\left(\frac{m}{a}\right)^2 + \left(\frac{n}{b}\right)^2\right]$

**应用领域**：
- 材料无损检测（NDT）
- 乐器声学设计
- 结构健康监测
- 工业质量控制（检测裂纹、缺陷）

**核心挑战**：从敲击产生的瞬态声信号中提取材料的共振频率特征，并实现材料的自动分类。

## 二、信号处理建模

### 处理流程

```
敲击声信号 (麦克风录音)
    ↓
预处理 (去直流、预加重)
    ↓
截取瞬态响应段
    ↓
FFT 频谱分析
    ↓
共振峰检测 (峰值提取)
    ↓
特征向量 → 材料分类
```

### 物理模型

敲击产生的声信号可建模为多个阻尼正弦波的叠加：

$$x(t) = \sum_{n=1}^{N} A_n \exp(-\gamma_n t) \sin(2\pi f_n t + \phi_n)$$

其中 $f_n$ 为第 $n$ 阶共振频率，$\gamma_n$ 为阻尼系数，$A_n$ 为振幅。

### 4.1 音频加载与预处理

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.io import wavfile
from scipy.signal import find_peaks, spectrogram
from scipy.fft import rfft, rfftfreq

# 设置中文字体
plt.rcParams['font.family'] = 'Noto Sans CJK SC'
plt.rcParams['font.sans-serif'] = ['Noto Sans CJK SC']
plt.rcParams['axes.unicode_minus'] = False

def load_audio(filepath):
    """加载音频文件"""
    sr, data = wavfile.read(filepath)
    if data.dtype == np.int16:
        data = data.astype(np.float32) / 32768.0
    elif data.dtype == np.int32:
        data = data.astype(np.float32) / 2147483648.0
    # 如果是立体声，取单声道
    if data.ndim > 1:
        data = data.mean(axis=1)
    return sr, data

def generate_synthetic_impact(sr=44100, duration=1.0, material='metal'):
    """生成合成敲击声（不同材料）"""
    t = np.arange(int(sr * duration)) / sr

    # 不同材料的共振参数 (频率Hz, 振幅, 阻尼系数)
    params = {
        'metal': [(2200, 0.8, 2.0), (4400, 0.5, 3.0), (6600, 0.3, 4.0),
                  (8800, 0.2, 5.0), (11000, 0.1, 6.0)],
        'glass': [(1500, 1.0, 1.0), (3000, 0.8, 1.5), (4500, 0.6, 2.0),
                  (6000, 0.4, 2.5), (7500, 0.3, 3.0)],
        'wood': [(300, 1.0, 8.0), (600, 0.6, 10.0), (900, 0.3, 12.0),
                 (1200, 0.15, 15.0)],
        'plastic': [(800, 0.7, 5.0), (1600, 0.4, 7.0), (2400, 0.2, 9.0),
                    (3200, 0.1, 12.0)],
        'ceramic': [(1800, 0.9, 2.5), (3600, 0.6, 3.5), (5400, 0.4, 4.5),
                    (7200, 0.25, 5.5)],
    }

    signal = np.zeros_like(t)
    for freq, amp, gamma in params[material]:
        signal += amp * np.exp(-gamma * t) * np.sin(2 * np.pi * freq * t)

    # 添加噪声
    signal += np.random.normal(0, 0.02, len(t))
    return t, signal

# 示例：生成不同材料的敲击声
sr = 44100
materials = ['metal', 'glass', 'wood', 'plastic', 'ceramic']
material_cn = {'metal': '金属', 'glass': '玻璃', 'wood': '木头',
               'plastic': '塑料', 'ceramic': '陶瓷'}

signals = {}
for mat in materials:
    t, sig = generate_synthetic_impact(sr=sr, material=mat)
    signals[mat] = sig

# 可视化波形
fig, axes = plt.subplots(5, 1, figsize=(14, 12), sharex=True)
for i, mat in enumerate(materials):
    axes[i].plot(t[:int(sr*0.2)], signals[mat][:int(sr*0.2)], 'b-', linewidth=0.3)
    axes[i].set_ylabel(f'{material_cn[mat]}\n振幅')
axes[4].set_xlabel('时间 (秒)')
plt.suptitle('不同材料的敲击声波形', fontsize=14)
plt.tight_layout()
plt.savefig('impact_waveforms.png', dpi=150, bbox_inches='tight')
plt.show()
```

### 4.2 FFT 频谱分析

```python
def compute_fft(signal, sr):
    """计算 FFT 频谱"""
    N = len(signal)
    yf = rfft(signal)
    xf = rfftfreq(N, 1/sr)
    magnitude = np.abs(yf) * 2.0 / N
    return xf, magnitude

# 计算并可视化频谱
fig, axes = plt.subplots(5, 1, figsize=(14, 12), sharex=True)
for i, mat in enumerate(materials):
    freqs, mag = compute_fft(signals[mat], sr)
    # 只显示 0-15 kHz
    mask = freqs <= 15000
    axes[i].plot(freqs[mask], mag[mask], 'b-', linewidth=0.5)
    axes[i].set_ylabel(f'{material_cn[mat]}\n幅度')
    axes[i].set_xlim(0, 15000)
axes[4].set_xlabel('频率 (Hz)')
```

### 4.3 共振峰检测

```python
def find_resonance_peaks(freqs, magnitude, n_peaks=5, min_distance=200, min_height_ratio=0.05):
    """
    检测共振峰
    """
    # 最小高度为最大值的一定比例
    min_height = magnitude.max() * min_height_ratio

    # 频率分辨率
    freq_resolution = freqs[1] - freqs[0]
    min_distance_points = int(min_distance / freq_resolution)

    peaks, properties = find_peaks(magnitude,
                                    height=min_height,
                                    distance=min_distance_points,
                                    prominence=min_height * 0.5)

    # 按幅度排序，取前 n_peaks 个
    if len(peaks) > 0:
        sorted_idx = np.argsort(magnitude[peaks])[::-1][:n_peaks]
        peaks = peaks[sorted_idx]
        peaks = np.sort(peaks)

    return peaks, freqs[peaks], magnitude[peaks]

# 对每种材料检测共振峰
fig, axes = plt.subplots(5, 1, figsize=(14, 12), sharex=True)
resonance_features = {}

for i, mat in enumerate(materials):
    freqs, mag = compute_fft(signals[mat], sr)
    mask = freqs <= 15000
    freqs_cut = freqs[mask]
    mag_cut = mag[mask]

    peak_idx, peak_freqs, peak_mags = find_resonance_peaks(freqs_cut, mag_cut)

    axes[i].plot(freqs_cut, mag_cut, 'b-', linewidth=0.5)
    axes[i].plot(peak_freqs, peak_mags, 'rv', markersize=8)
    axes[i].set_ylabel(f'{material_cn[mat]}\n幅度')
    axes[i].set_xlim(0, 15000)

    # 标注频率
    for f, m in zip(peak_freqs, peak_mags):
        axes[i].annotate(f'{f:.0f}Hz', xy=(f, m), xytext=(0, 10),
                         textcoords='offset points', fontsize=7, ha='center')

    resonance_features[mat] = peak_freqs
    print(f"{material_cn[mat]}: 共振频率 = {peak_freqs.astype(int)} Hz")

axes[4].set_xlabel('频率 (Hz)')
plt.suptitle('共振峰检测结果', fontsize=14)
plt.tight_layout()
plt.savefig('resonance_peaks.png', dpi=150, bbox_inches='tight')
```

### 4.4 频谱包络与倒谱分析

```python
def compute_cepstrum(signal, sr):
    """计算倒谱（cepstrum）用于共振峰提取"""
    # FFT
    spectrum = np.fft.fft(signal)
    # 对数幅度谱
    log_mag = np.log(np.abs(spectrum) + 1e-10)
    # 逆 FFT 得到倒谱
    cepstrum = np.fft.ifft(log_mag).real
    quefrency = np.arange(len(cepstrum)) / sr
    return quefrency, cepstrum

# 倒谱分析
fig, axes = plt.subplots(2, 1, figsize=(14, 8))
for mat in ['metal', 'glass']:
    quef, cep = compute_cepstrum(signals[mat], sr)
    # 只显示前 10ms
    mask = quef <= 0.01
    axes[0].plot(quef[mask]*1000, cep[mask], label=material_cn[mat])
axes[0].set_xlabel('倒频率 (ms)')
axes[0].set_ylabel('倒谱幅度')
axes[0].set_title('倒谱分析')
axes[0].legend()

# 短时频谱图
for i, mat in enumerate(['metal']):
    f, t_spec, Sxx = spectrogram(signals[mat], sr, nperseg=1024, noverlap=512)
    mask_f = f <= 10000
    axes[1].pcolormesh(t_spec*1000, f[mask_f], 10*np.log10(Sxx[mask_f]+1e-10),
                        cmap='hot', shading='auto')
    axes[1].set_xlabel('时间 (ms)')
    axes[1].set_ylabel('频率 (Hz)')
    axes[1].set_title('金属敲击声 - 短时频谱图')
```

### 4.5 材料分类

```python
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report

def extract_material_features(signal, sr, n_peaks=10):
    """提取材料声学特征"""
    freqs, mag = compute_fft(signal, sr)
    mask = (freqs >= 100) & (freqs <= 15000)
    freqs_cut = freqs[mask]
    mag_cut = mag[mask]

    peak_idx, peak_freqs, peak_mags = find_resonance_peaks(freqs_cut, mag_cut, n_peaks=n_peaks)

    features = []
    # 前 n_peaks 个共振频率
    for j in range(n_peaks):
        if j < len(peak_freqs):
            features.extend([peak_freqs[j], peak_mags[j]])
        else:
            features.extend([0, 0])

    # 频谱质心
    spectral_centroid = np.sum(freqs_cut * mag_cut) / (np.sum(mag_cut) + 1e-10)
    features.append(spectral_centroid)

    # 频谱带宽
    spectral_bw = np.sqrt(np.sum(((freqs_cut - spectral_centroid)**2) * mag_cut) / (np.sum(mag_cut) + 1e-10))
    features.append(spectral_bw)

    # 频谱衰减
    spectral_rolloff_idx = np.where(np.cumsum(mag_cut) >= 0.85 * np.sum(mag_cut))[0]
    if len(spectral_rolloff_idx) > 0:
        features.append(freqs_cut[spectral_rolloff_idx[0]])
    else:
        features.append(0)

    return np.array(features)

# 生成训练数据
n_samples_per_class = 50
X_all = []
y_all = []

for mat_idx, mat in enumerate(materials):
    for _ in range(n_samples_per_class):
        # 添加随机变化
        t, sig = generate_synthetic_impact(sr=sr, material=mat)
        # 随机偏移和缩放
        sig = sig * np.random.uniform(0.5, 1.5)
        features = extract_material_features(sig, sr)
        X_all.append(features)
        y_all.append(mat_idx)

X_all = np.array(X_all)
y_all = np.array(y_all)

# 训练分类器
X_train, X_test, y_train, y_test = train_test_split(X_all, y_all, test_size=0.3, random_state=42)

clf = RandomForestClassifier(n_estimators=100, random_state=42)
clf.fit(X_train, y_train)
y_pred = clf.predict(X_test)

print("材料分类报告:")
print(classification_report(y_test, y_pred, target_names=[material_cn[m] for m in materials]))

# 特征重要性
feature_names = []
for j in range(10):
    feature_names.extend([f'频率{j+1}', f'幅度{j+1}'])
feature_names.extend(['频谱质心', '频谱带宽', '频谱衰减'])

importances = clf.feature_importances_
top_idx = np.argsort(importances)[::-1][:10]
print("\n最重要的特征:")
for idx in top_idx:
    print(f"  {feature_names[idx]}: {importances[idx]:.4f}")
```
