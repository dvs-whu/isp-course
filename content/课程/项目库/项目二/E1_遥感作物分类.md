## 一、项目背景

精准农业（Precision Agriculture）是现代农业发展的重要方向。通过遥感技术监测农田作物类型和生长状态，可以实现：

### 应用场景
- **作物种植结构调查**：了解区域作物种植比例，指导农业政策
- **产量预估**：结合作物类型和生长参数预测粮食产量
- **病虫害监测**：不同作物的光谱响应差异可指示健康状态变化
- **资源优化**：根据作物类型精准灌溉、施肥

### 信号处理视角
遥感图像本质上是**多维信号**——每个像素包含多个光谱波段的反射率值，构成一个多维光谱信号向量。信号处理技术在其中的应用：
- **频域特征提取**：利用 Gabor 滤波器捕捉纹理的频率和方向信息
- **降维**：通过 PCA 等方法减少冗余波段，提取主要信息
- **滤波器组设计**：设计针对不同作物光谱特征的特征提取器

## 二、信号处理建模

### 2.1 处理流程

```
多光谱遥感图像 (M×N×B，B个波段)
  │
  ▼
预处理
  ├── 辐射校正（DN值 → 反射率）
  ├── 大气校正
  └── 几何校正
  │
  ▼
特征提取
  ├── 光谱特征：各波段反射率、植被指数(NDVI)
  ├── 空间特征：Gabor 滤波器组响应
  └── 纹理特征：GLCM 灰度共生矩阵
  │
  ▼
降维
  ├── PCA 主成分分析
  └── 最佳波段选择
  │
  ▼
分类器
  ├── CNN（空间-光谱联合特征）
  └── SVM（手工特征 + 核方法）
  │
  ▼
作物分类图 (M×N×1)
```

### 2.2 关键技术原理

**Gabor 滤波器**（频域纹理分析核心）：
$$g(x,y;\lambda,\theta,\psi,\sigma,\gamma) = \exp\left(-\frac{x'^2+\gamma^2 y'^2}{2\sigma^2}\right) \exp\left(j\left(2\pi\frac{x'}{\lambda}+\psi\right)\right)$$
其中 $x'=x\cos\theta+y\sin\theta$，$y'=-x\sin\theta+y\cos\theta$

**NDVI 归一化植被指数**：
$$NDVI = \frac{NIR - RED}{NIR + RED}$$

**PCA 降维**：
$$Z = XW$$
其中 $W$ 为协方差矩阵的特征向量矩阵

### 4.2 多光谱特征提取

```python
def compute_ndvi(image, red_idx=3, nir_idx=7):
    """
    计算 NDVI 植被指数
    Sentinel-2 波段：B4=Red(索引3), B8=NIR(索引7)
    """
    red = image[:, :, red_idx].astype(np.float64)
    nir = image[:, :, nir_idx].astype(np.float64)
    
    ndvi = (nir - red) / (nir + red + 1e-10)
    return ndvi

def compute_spectral_indices(image):
    """计算多种光谱指数"""
    indices = {}
    
    # NDVI - 归一化植被指数
    indices['NDVI'] = compute_ndvi(image)
    
    # NDWI - 归一化水体指数 (Green - NIR) / (Green + NIR)
    green = image[:, :, 2].astype(np.float64)
    nir = image[:, :, 7].astype(np.float64)
    indices['NDWI'] = (green - nir) / (green + nir + 1e-10)
    
    # NDBI - 归一化建筑指数 (SWIR - NIR) / (SWIR + NIR)
    swir = image[:, :, 11].astype(np.float64)
    indices['NDBI'] = (swir - nir) / (swir + nir + 1e-10)
    
    return indices

def visualize_multispectral(image, class_name=""):
    """可视化多光谱图像的各波段和光谱指数"""
    fig, axes = plt.subplots(2, 4, figsize=(16, 8))
    
    # 显示关键波段
    band_names = ['B2-蓝', 'B3-绿', 'B4-红', 'B8-近红外',
                  'B5-红边', 'B6-红边', 'B11-短波红外', 'B12-短波红外']
    band_indices = [1, 2, 3, 7, 4, 5, 11, 12]
    
    for i, (name, idx) in enumerate(zip(band_names, band_indices)):
        ax = axes[i // 4, i % 4]
        if idx < image.shape[-1]:
            ax.imshow(image[:, :, idx], cmap='gray')
        ax.set_title(name, fontproperties=font_prop)

    plt.tight_layout()
    plt.savefig(f'multispectral_{class_name}.png', dpi=150, bbox_inches='tight')
```

### 4.3 Gabor 纹理特征提取

```python
from scipy.ndimage import convolve

def gabor_filter_bank(frequencies=[0.1, 0.2, 0.3, 0.4],
                      orientations=[0, np.pi/4, np.pi/2, 3*np.pi/4],
                      sigma_x=3, sigma_y=3, size=21):
    """
    创建 Gabor 滤波器组
    用于提取遥感图像的空间纹理特征
    """
    filters = []
    
    for freq in frequencies:
        for theta in orientations:
            # 创建网格
            x = np.arange(size) - size // 2
            y = np.arange(size) - size // 2
            X, Y = np.meshgrid(x, y)
            
            # 旋转坐标
            X_rot = X * np.cos(theta) + Y * np.sin(theta)
            Y_rot = -X * np.sin(theta) + Y * np.cos(theta)
            
            # Gabor 函数
            gaussian = np.exp(-(X_rot**2 / (2*sigma_x**2) + Y_rot**2 / (2*sigma_y**2)))
            sinusoid = np.cos(2 * np.pi * freq * X_rot)
            gabor = gaussian * sinusoid
            
            filters.append({
                'kernel': gabor,
                'frequency': freq,
                'orientation': theta * 180 / np.pi
            })
    
    return filters

def extract_gabor_features(image, filters):
    """用 Gabor 滤波器组提取纹理特征"""
    if len(image.shape) == 3:
        # 对多波段图像取均值
        image = np.mean(image, axis=-1)
    
    image = image.astype(np.float64)
    
    features = []
    responses = []
    
    for f in filters:
        response = convolve(image, f['kernel'])
        # 统计特征：均值和标准差
        features.append(np.mean(np.abs(response)))
        features.append(np.std(response))
        responses.append(response)
    
    return np.array(features), responses

def visualize_gabor_responses(image, filters, n_show=8):
    """可视化 Gabor 滤波器响应"""
    fig, axes = plt.subplots(2, n_show//2, figsize=(16, 6))
    
    _, responses = extract_gabor_features(image, filters)
    
    for i in range(n_show):
        ax = axes[i // (n_show//2), i % (n_show//2)]
        ax.imshow(np.abs(responses[i]), cmap='hot')
        f = filters[i]
        ax.set_title(f'f={f["frequency"]:.1f} θ={f["orientation"]:.0f}°',
                     fontproperties=font_prop, fontsize=9)
        ax.axis('off')
    
    plt.tight_layout()
    plt.savefig('gabor_responses.png', dpi=150, bbox_inches='tight')
    plt.show()
```

### 4.4 PCA 降维

```python
def pca_analysis(images, n_components=10):
    """
    PCA 降维分析
    
    对多光谱图像进行主成分分析，保留主要信息
    """
    # 展平所有像素
    n_samples, h, w, n_bands = images.shape
    X = images.reshape(-1, n_bands)
    
    # 标准化
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    
    # PCA
    pca = PCA(n_components=n_components)
    X_pca = pca.fit_transform(X_scaled)
    
    # 解释方差比
    explained_var = pca.explained_variance_ratio_
    cumulative_var = np.cumsum(explained_var)
    
    fig, axes = plt.subplots(1, 2, figsize=(12, 4))
    
    # 各主成分方差
    axes[0].bar(range(1, n_components+1), explained_var, alpha=0.7, label='单独')
    axes[0].plot(range(1, n_components+1), cumulative_var, 'ro-', label='累积')
    axes[0].set_xlabel('主成分', fontproperties=font_prop)
    axes[0].set_ylabel('解释方差比', fontproperties=font_prop)
    axes[0].set_title('PCA 解释方差', fontproperties=font_prop)
    axes[0].legend(prop=font_prop)
    
    # 波段贡献（loading）
    loadings = pca.components_[:3]
    axes[1].imshow(loadings, cmap='RdBu_r', aspect='auto')
    axes[1].set_xlabel('原始波段', fontproperties=font_prop)
    axes[1].set_ylabel('主成分', fontproperties=font_prop)
    axes[1].set_title('主成分载荷', fontproperties=font_prop)
    
    plt.tight_layout()
    plt.savefig('pca_analysis.png', dpi=150, bbox_inches='tight')
    plt.show()
    
    return pca, X_pca
```

### 4.5 CNN 作物分类模型

```python
import tensorflow as tf
from tensorflow.keras import layers, models

def build_crop_classifier(input_shape=(64, 64, 3), num_classes=10):
    """构建作物分类 CNN"""
    model = models.Sequential([
        # 卷积块 1
        layers.Conv2D(32, 3, padding='same', input_shape=input_shape),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.Conv2D(32, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        layers.Dropout(0.25),
        
        # 卷积块 2
        layers.Conv2D(64, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.Conv2D(64, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        layers.Dropout(0.25),
        
        # 卷积块 3
        layers.Conv2D(128, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.GlobalAveragePooling2D(),
        
        # 分类头
        layers.Dense(128, activation='relu'),
        layers.Dropout(0.5),
        layers.Dense(num_classes, activation='softmax')
    ])
    
    return model

def train_classifier(dataset, class_names, epochs=30, batch_size=64):
    """训练作物分类器"""
    # 数据预处理
    def preprocess(image, label):
        image = tf.cast(image, tf.float32) / 255.0
        return image, label
    
    train_ds = dataset['train'].map(preprocess).shuffle(1000).batch(batch_size).prefetch(tf.data.AUTOTUNE)
    
    model = build_crop_classifier(num_classes=len(class_names))
    model.compile(
        optimizer='adam',
        loss='sparse_categorical_crossentropy',
        metrics=['accuracy']
    )
    
    history = model.fit(train_ds, epochs=epochs)
    
    return model, history
```

### 4.6 SVM 分类（手工特征）

```python
from sklearn.svm import SVC
from sklearn.metrics import classification_report, confusion_matrix

def svm_classify_with_features(images, labels, class_names):
    """基于手工特征的 SVM 分类"""
    # 提取特征
    gabor_filters = gabor_filter_bank()
    features_list = []
    
    for img in images:
        # Gabor 纹理特征
        gabor_feat, _ = extract_gabor_features(img, gabor_filters)
        
        # 光谱统计特征
        spectral_mean = np.mean(img, axis=(0, 1))
        spectral_std = np.std(img, axis=(0, 1))
        
        # 组合特征
        feat = np.concatenate([gabor_feat, spectral_mean, spectral_std])
        features_list.append(feat)
    
    X = np.array(features_list)
    
    # 标准化
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    
    # PCA 降维
    pca = PCA(n_components=min(50, X_scaled.shape[1]))
    X_pca = pca.fit_transform(X_scaled)
    
    # 训练 SVM
    clf = SVC(kernel='rbf', C=10, gamma='scale')
    clf.fit(X_pca, labels)
    
    # 评估
    y_pred = clf.predict(X_pca)
    print(classification_report(labels, y_pred, target_names=class_names))
    
    return clf, scaler, pca
```
