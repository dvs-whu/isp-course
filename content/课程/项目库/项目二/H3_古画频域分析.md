## 一、项目背景

中国古代绘画是中华文明的瑰宝，承载着丰富的美学思想和历史信息。随着数字化技术的发展，高精度扫描和摄影使得我们能够以数字信号的视角重新审视这些艺术作品。

### 核心问题
- 古画在流传过程中会出现**霉斑、裂纹、褪色**等退化现象
- 修复专家需要识别画面的**纹理方向**和**笔触特征**以指导修复
- 频域分析能够揭示肉眼难以察觉的**周期性纹理**和**隐藏信息**

### 信号处理视角
一幅绘画图像本质上是一个二维信号 $f(x, y)$。通过二维傅里叶变换，我们可以：
- 在频域中分析画面的**纹理频率分布**
- 识别画面中的**周期性噪声**（如扫描条纹、印刷网点）
- 根据频域特征进行**智能去噪**和**增强**

## 二、信号处理建模

### 2.1 处理流程

```
古画数字图像 f(x, y)
  │
  ▼
灰度化 / 通道分离
  │
  ▼
二维 FFT: F(u, v) = ∫∫ f(x,y) e^{-j2π(ux+vy)} dx dy
  │
  ├── 功率谱 |F(u,v)|²  ── 能量分布分析
  ├── 径向平均 PSD  ── 频率衰减特性
  └── 角度分布  ── 纹理方向性
  │
  ▼
频域滤波 / 增强
  ├── 带阻滤波 ── 去除周期性噪声
  ├── 方向滤波 ── 提取特定方向笔触
  └── 同态滤波 ── 光照不均匀校正
  │
  ▼
逆 FFT → 修复 / 增强后的图像
```

### 2.2 关键数学原理

**二维离散傅里叶变换**：
$$F(u,v) = \sum_{x=0}^{M-1}\sum_{y=0}^{N-1} f(x,y) e^{-j2\pi\left(\frac{ux}{M}+\frac{vy}{N}\right)}$$

**径向功率谱密度**：
将频谱按极坐标表示，对相同半径 $r$ 的频率分量求平均：
$$PSD(r) = \frac{1}{2\pi r} \oint |F(r,\theta)|^2 d\theta$$

**方向能量分布**：
对相同角度 $\theta$ 的频率分量积分：
$$E(\theta) = \int_0^{R} |F(r,\theta)|^2 dr$$

### 4.1 图像频域分析

```python
import numpy as np
import matplotlib.pyplot as plt
from matplotlib import font_manager
from PIL import Image
from scipy import fft

# 设置中文字体
font_path = '/usr/share/fonts/truetype/noto/NotoSansCJKsc-Regular.otf'
font_prop = font_manager.FontProperties(fname=font_path)
plt.rcParams['font.family'] = font_prop.get_name()
plt.rcParams['axes.unicode_minus'] = False

def load_painting(image_path):
    """加载古画图像并转为灰度"""
    img = Image.open(image_path).convert('L')
    img_array = np.array(img, dtype=np.float64)
    return img_array

def compute_2d_fft(image):
    """计算二维FFT并居中"""
    F = fft.fft2(image)
    F_shifted = fft.fftshift(F)  # 将零频移到中心
    magnitude = np.abs(F_shifted)
    phase = np.angle(F_shifted)
    return F_shifted, magnitude, phase

def visualize_frequency_domain(image, magnitude, phase):
    """可视化原始图像和频域"""
    fig, axes = plt.subplots(1, 3, figsize=(18, 5))
    
    # 原始图像
    axes[0].imshow(image, cmap='gray')
    axes[0].set_title('原始古画', fontproperties=font_prop)
    axes[0].axis('off')
    
    # 功率谱（对数尺度）
    power_spectrum = np.log1p(magnitude)
    axes[1].imshow(power_spectrum, cmap='hot')
    axes[1].set_title('频谱幅度（对数尺度）', fontproperties=font_prop)
    axes[1].axis('off')
    
    # 相位谱
    axes[2].imshow(phase, cmap='hsv')
    axes[2].set_title('相位谱', fontproperties=font_prop)
    axes[2].axis('off')
    
    plt.tight_layout()
    plt.savefig('painting_frequency_domain.png', dpi=150, bbox_inches='tight')
    plt.show()
```

### 4.2 径向平均功率谱分析

```python
def radial_average_psd(magnitude, n_bins=100):
    """计算径向平均功率谱密度"""
    rows, cols = magnitude.shape
    center_r, center_c = rows // 2, cols // 2
    
    # 创建距离矩阵
    y, x = np.ogrid[:rows, :cols]
    radius = np.sqrt((x - center_c)**2 + (y - center_r)**2).astype(int)
    
    # 径向平均
    max_radius = min(center_r, center_c)
    psd_radial = np.zeros(max_radius)
    
    for r in range(max_radius):
        mask = (radius == r)
        if np.sum(mask) > 0:
            psd_radial[r] = np.mean(magnitude[mask]**2)
    
    return psd_radial

def analyze_texture_frequency(image, title="古画"):
    """完整频域分析：径向PSD + 方向分布"""
    F_shifted, magnitude, _ = compute_2d_fft(image)
    
    fig, axes = plt.subplots(1, 3, figsize=(18, 5))
    
    # 原始图像
    axes[0].imshow(image, cmap='gray')
    axes[0].set_title(f'{title} - 原图', fontproperties=font_prop)
    axes[0].axis('off')
    
    # 径向平均 PSD
    psd = radial_average_psd(magnitude)
    freqs = np.arange(len(psd))
    axes[1].loglog(freqs[1:], psd[1:], 'b-', linewidth=1.5)
    axes[1].set_xlabel('空间频率 (cycles/pixel)', fontproperties=font_prop)
    axes[1].set_ylabel('功率谱密度', fontproperties=font_prop)
    axes[1].set_title('径向平均功率谱', fontproperties=font_prop)
    axes[1].grid(True, alpha=0.3)
    
    # 方向能量分布
    rows, cols = magnitude.shape
    center_r, center_c = rows // 2, cols // 2
    y, x = np.ogrid[:rows, :cols]
    angles = np.arctan2(y - center_r, x - center_c)
    
    n_angles = 36
    angle_bins = np.linspace(-np.pi, np.pi, n_angles + 1)
    direction_energy = np.zeros(n_angles)
    
    for i in range(n_angles):
        mask = (angles >= angle_bins[i]) & (angles < angle_bins[i+1])
        if np.sum(mask) > 0:
            direction_energy[i] = np.mean(magnitude[mask]**2)
    
    theta = (angle_bins[:-1] + angle_bins[1:]) / 2 * 180 / np.pi
    axes[2].bar(theta, direction_energy, width=10, alpha=0.7, color='coral')
    axes[2].set_xlabel('角度 (度)', fontproperties=font_prop)
    axes[2].set_ylabel('平均能量', fontproperties=font_prop)
    axes[2].set_title('方向能量分布', fontproperties=font_prop)
    
    plt.tight_layout()
    plt.savefig(f'{title}_texture_analysis.png', dpi=150, bbox_inches='tight')
    plt.show()
```

### 4.3 频域去噪与增强

```python
def remove_periodic_noise(image, noise_centers, radius=10):
    """去除周期性噪声（如扫描条纹）"""
    F_shifted, magnitude, phase = compute_2d_fft(image)
    
    rows, cols = image.shape
    y, x = np.ogrid[:rows, :cols]
    
    # 创建带阻滤波器
    mask = np.ones((rows, cols), dtype=np.float64)
    for (cx, cy) in noise_centers:
        dist = np.sqrt((x - cx)**2 + (y - cy)**2)
        mask[dist <= radius] = 0
    
    # 应用滤波器
    F_filtered = F_shifted * mask
    F_ishifted = fft.ifftshift(F_filtered)
    result = np.real(fft.ifft2(F_ishifted))
    
    return result, mask

def directional_filter(image, angle_range, band_type='bandpass'):
    """方向滤波：提取特定方向的纹理"""
    F_shifted, magnitude, phase = compute_2d_fft(image)
    
    rows, cols = image.shape
    center_r, center_c = rows // 2, cols // 2
    y, x = np.ogrid[:rows, :cols]
    angles = np.arctan2(y - center_r, x - center_c) * 180 / np.pi
    
    # 创建方向滤波器
    mask = np.zeros((rows, cols), dtype=np.float64)
    angle_low, angle_high = angle_range
    
    # 同时保留正负方向（对称）
    for sign in [1, -1]:
        a_low = angle_low * sign
        a_high = angle_high * sign
        a_min, a_max = min(a_low, a_high), max(a_low, a_high)
        mask[(angles >= a_min) & (angles <= a_max)] = 1.0
    
    # 平滑过渡（避免振铃效应）
    from scipy.ndimage import gaussian_filter
    mask = gaussian_filter(mask, sigma=2)
    
    if band_type == 'bandstop':
        mask = 1 - mask
    
    F_filtered = F_shifted * mask
    F_ishifted = fft.ifftshift(F_filtered)
    result = np.real(fft.ifft2(F_ishifted))
    
    return result, mask

def homomorphic_filter(image, gamma_l=0.5, gamma_h=2.0, cutoff=30):
    """同态滤波：校正光照不均匀"""
    img_log = np.log1p(image.astype(np.float64))
    
    F = fft.fft2(img_log)
    F_shifted = fft.fftshift(F)
    
    rows, cols = image.shape
    y, x = np.ogrid[:rows, :cols]
    center_r, center_c = rows // 2, cols // 2
    dist = np.sqrt((x - center_c)**2 + (y - center_r)**2)
    
    # Butterworth 高通滤波器
    H = (gamma_h - gamma_l) * (1 - np.exp(-(dist**2) / (2 * cutoff**2))) + gamma_l
    
    F_filtered = F_shifted * H
    result = np.real(fft.ifft2(fft.ifftshift(F_filtered)))
    result = np.expm1(result)
    result = np.clip(result, 0, 255).astype(np.uint8)
    
    return result
```

### 4.4 对比不同朝代绘画的频域特征

```python
def compare_dynasty_paintings(paintings_dict):
    """对比不同时期绘画的频域特性"""
    fig, axes = plt.subplots(2, len(paintings_dict), figsize=(5*len(paintings_dict), 10))
    
    psd_dict = {}
    for idx, (name, img_path) in enumerate(paintings_dict.items()):
        img = load_painting(img_path)
        _, magnitude, _ = compute_2d_fft(img)
        
        # 上排：原图
        axes[0, idx].imshow(img, cmap='gray')
        axes[0, idx].set_title(name, fontproperties=font_prop)
        axes[0, idx].axis('off')
        
        # 下排：径向PSD
        psd = radial_average_psd(magnitude)
        psd_dict[name] = psd
        freqs = np.arange(1, len(psd))
        axes[1, idx].loglog(freqs, psd[1:], 'b-', linewidth=1)
        axes[1, idx].set_xlabel('频率', fontproperties=font_prop)
        axes[1, idx].set_ylabel('PSD', fontproperties=font_prop)
        axes[1, idx].grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.savefig('dynasty_comparison.png', dpi=150, bbox_inches='tight')
    plt.show()
    
    return psd_dict
```
