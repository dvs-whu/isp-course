## 三、实验原理

### 3.1 事件相机简介

事件相机（如DVS、DAVIS）是一种**异步**视觉传感器。与传统帧相机不同，每个像素**独立**地检测亮度变化：

- 当像素 $(x, y)$ 的对数亮度变化超过阈值 $C$ 时，产生一个事件：
$$\log I(x, y, t) - \log I(x, y, t - \Delta t) \geq \pm C$$
- 每个事件表示为四元组：$\mathbf{e} = (x, y, t, p)$
  - $x, y$：像素坐标
  - $t$：时间戳（微秒级精度）
  - $p \in \{+1, -1\}$：极性（亮度增加/减少）

### 3.2 事件相机的优势

| 特性 | 传统帧相机 | 事件相机 |
|------|-----------|---------|
| 时间分辨率 | 30-120 fps | 微秒级（>100kHz等效） |
| 动态范围 | 60-70 dB | 120-140 dB |
| 数据冗余 | 高（全帧传输） | 低（仅变化像素） |
| 运动模糊 | 有 | 几乎无 |
| 功耗 | 较高 | 极低（mW级） |

### 3.3 事件到帧转换

将事件流转换为固定时间窗口内的"事件帧"是常用的数据处理方法：

**时间窗口法**：在时间窗口 $\Delta T$ 内，统计每个像素的事件数量或极性累加值：

$$I_{\text{event}}(x, y) = \sum_{t_k \in [T, T+\Delta T]} p_k \cdot \delta(x - x_k, y - y_k)$$

**时间表面法（Time Surface）**：记录每个像素最近一次事件的时间戳：

$$S(x, y) = \exp\left(-\frac{t_{\text{now}} - t_{\text{last}}(x, y)}{\tau}\right)$$

其中 $\tau$ 为衰减时间常数。

```python
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.colors import LinearSegmentedColormap

plt.rcParams['font.sans-serif'] = ['SimHei', 'Arial Unicode MS']
plt.rcParams['axes.unicode_minus'] = False

np.random.seed(42)

# 事件相机参数
SENSOR_WIDTH = 240
SENSOR_HEIGHT = 180

def generate_synthetic_events(duration_ms=100, num_events=50000):
    """
    模拟事件相机数据：一个移动的亮斑在暗背景上运动
    返回: events (N, 4) 数组, 每行为 (t, x, y, p)
    """
    events = []
    t_us = np.linspace(0, duration_ms * 1000, num_events)  # 微秒

    # 圆心随时间移动（椭圆轨迹）
    cx = SENSOR_WIDTH / 2 + 40 * np.cos(2 * np.pi * t_us / (duration_ms * 1000))
    cy = SENSOR_HEIGHT / 2 + 30 * np.sin(2 * np.pi * t_us / (duration_ms * 1000))

    # 圆半径
    radius = 20

    for i in range(num_events):
        # 在圆心附近随机生成事件
        angle = np.random.uniform(0, 2 * np.pi)
        r = radius * np.sqrt(np.random.uniform(0, 1))
        x = int(cx[i] + r * np.cos(angle))
        y = int(cy[i] + r * np.sin(angle))

        # 边界检查
        if 0 <= x < SENSOR_WIDTH and 0 <= y < SENSOR_HEIGHT:
            # 极性：圆内为正（变亮），圆边界为负（变暗）
            dist = np.sqrt((x - cx[i]) ** 2 + (y - cy[i]) ** 2)
            polarity = 1 if dist < radius * 0.7 else -1
            events.append([t_us[i], x, y, polarity])

    events = np.array(events)
    # 按时间排序
    events = events[events[:, 0].argsort()]
    print(f"生成事件数: {len(events)}")
    print(f"时间范围: [{events[0, 0]:.0f}, {events[-1, 0]:.0f}] μs")
    print(f"空间范围: x=[{events[:, 1].min():.0f}, {events[:, 1].max():.0f}], "
          f"y=[{events[:, 2].min():.0f}, {events[:, 2].max():.0f}]")
    print(f"正极性事件: {np.sum(events[:, 3] > 0)}, 负极性事件: {np.sum(events[:, 3] < 0)}")
    return events

events = generate_synthetic_events(duration_ms=100, num_events=50000)
```

### 步骤3：事件到帧转换——时间窗口法

```python
def events_to_frame(events, sensor_size, time_window_us=None, method='count'):
    """
    将事件流转换为帧图像
    参数:
        events       : (N, 4) 数组 [t, x, y, p]
        sensor_size  : (height, width)
        time_window_us : 时间窗口(微秒), None则使用全部事件
        method       : 'count'(计数) 或 'polarity'(极性累加)
    返回:
        frame : (H, W) 事件帧
    """
    H, W = sensor_size
    if time_window_us is not None:
        t_start = events[0, 0]
        mask = (events[:, 0] >= t_start) & (events[:, 0] < t_start + time_window_us)
        ev = events[mask]
    else:
        ev = events

    frame = np.zeros((H, W), dtype=np.float64)
    x = ev[:, 1].astype(int)
    y = ev[:, 2].astype(int)
    p = ev[:, 3]

    # 边界过滤
    valid = (x >= 0) & (x < W) & (y >= 0) & (y < H)

    if method == 'count':
        np.add.at(frame, (y[valid], x[valid]), np.abs(p[valid]))
    elif method == 'polarity':
        np.add.at(frame, (y[valid], x[valid]), p[valid])

    return frame

# 不同时间窗口的事件帧
```

### 步骤4：时间表面（Time Surface）

```python
def compute_time_surface(events, sensor_size, tau_us=5000):
    """
    计算时间表面
    参数:
        events    : (N, 4) 数组
        sensor_size : (H, W)
        tau_us    : 衰减时间常数(微秒)
    返回:
        surface_pos : 正极性时间表面
        surface_neg : 负极性时间表面
    """
    H, W = sensor_size
    t_last_pos = np.full((H, W), -np.inf)
    t_last_neg = np.full((H, W), -np.inf)

    t_now = events[-1, 0]

    for e in events:
        t, x, y, p = int(e[0]), int(e[1]), int(e[2]), int(e[3])
        if 0 <= x < W and 0 <= y < H:
            if p > 0:
                t_last_pos[y, x] = t
            else:
                t_last_neg[y, x] = t

    surface_pos = np.exp(-(t_now - t_last_pos) / tau_us)
    surface_pos[t_last_pos == -np.inf] = 0

    surface_neg = np.exp(-(t_now - t_last_neg) / tau_us)
    surface_neg[t_last_neg == -np.inf] = 0

    return surface_pos, surface_neg

# 计算不同时间常数的时间表面
tau_values = [1000, 5000, 20000]  # 微秒
fig, axes = plt.subplots(2, len(tau_values), figsize=(14, 6))

for idx, tau in enumerate(tau_values):
    s_pos, s_neg = compute_time_surface(events, (SENSOR_HEIGHT, SENSOR_WIDTH), tau)
    axes[0, idx].imshow(s_pos, cmap='Reds', vmin=0, vmax=1)
    axes[0, idx].set_title(f'正极性 τ={tau/1000:.0f}ms')
    axes[0, idx].axis('off')
    axes[1, idx].imshow(s_neg, cmap='Blues', vmin=0, vmax=1)
    axes[1, idx].set_title(f'负极性 τ={tau/1000:.0f}ms')
    axes[1, idx].axis('off')

plt.suptitle('时间表面（Time Surface）', fontsize=14)
plt.tight_layout()
```

### 步骤5：滑动事件帧与运动检测

```python
def sliding_event_reconstruction(events, sensor_size, window_us=10000,
                                  step_us=5000, method='polarity'):
    """
    滑动窗口重建事件帧序列
    """
    H, W = sensor_size
    t_start = events[0, 0]
    t_end = events[-1, 0]

    frames = []
    timestamps = []

    t_current = t_start
    while t_current + window_us <= t_end:
        mask = (events[:, 0] >= t_current) & (events[:, 0] < t_current + window_us)
        ev = events[mask]

        frame = np.zeros((H, W), dtype=np.float64)
        if len(ev) > 0:
            x = ev[:, 1].astype(int)
            y = ev[:, 2].astype(int)
            p = ev[:, 3]
            valid = (x >= 0) & (x < W) & (y >= 0) & (y < H)
            if method == 'polarity':
                np.add.at(frame, (y[valid], x[valid]), p[valid])
            else:
                np.add.at(frame, (y[valid], x[valid]), np.abs(p[valid]))

        frames.append(frame)
        timestamps.append(t_current + window_us / 2)
        t_current += step_us

    return frames, timestamps

frames, timestamps = sliding_event_reconstruction(
    events, (SENSOR_HEIGHT, SENSOR_WIDTH),
    window_us=10000, step_us=5000
)

plt.suptitle('滑动窗口事件帧序列', fontsize=13)
```
