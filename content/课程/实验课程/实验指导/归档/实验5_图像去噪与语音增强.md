# 实验5：图像去噪与语音增强

## 一、实验目的

1. 掌握二维傅里叶变换（2D-FFT）在图像频域去噪中的应用
2. 理解频域滤波器（低通、带阻）的设计与实现
3. 学习谱减法（Spectral Subtraction）语音增强的基本原理
4. 对比不同去噪/增强方法的性能

## 三、实验原理

### 3.1 二维离散傅里叶变换（2D-DFT）

图像 $f(x, y)$ 的二维DFT定义为：

$$F(u, v) = \sum_{x=0}^{M-1}\sum_{y=0}^{N-1} f(x, y) \cdot e^{-j2\pi(ux/M + vy/N)}$$

逆变换为：

$$f(x, y) = \frac{1}{MN}\sum_{u=0}^{M-1}\sum_{v=0}^{N-1} F(u, v) \cdot e^{j2\pi(ux/M + vy/N)}$$

频谱中，**低频分量**对应图像的平滑区域，**高频分量**对应边缘和细节。噪声通常表现为高频能量的增加。

### 3.2 频域去噪方法

- **理想低通滤波器（ILPF）**：$H(u,v) = \begin{cases} 1, & D(u,v) \leq D_0 \\ 0, & D(u,v) > D_0 \end{cases}$
- **巴特沃斯低通滤波器（BLPF）**：$H(u,v) = \frac{1}{1 + [D(u,v)/D_0]^{2n}}$
- **高斯低通滤波器（GLPF）**：$H(u,v) = e^{-D^2(u,v)/(2D_0^2)}$

其中 $D(u,v) = \sqrt{(u-M/2)^2 + (v-N/2)^2}$ 为频率点到频谱中心的距离。

### 3.3 谱减法语音增强

含噪语音的短时傅里叶变换（STFT）：

$$X(k, l) = S(k, l) + N(k, l)$$

谱减法假设噪声功率谱已知（通常在静音段估计）：

$$|\hat{S}(k,l)|^2 = |X(k,l)|^2 - \alpha|\hat{N}(k)|^2$$

为防止出现负值，加入半波整流：

$$|\hat{S}(k,l)| = \max\left(\sqrt{|X(k,l)|^2 - \alpha|\hat{N}(k)|^2},\ \beta\cdot|X(k,l)|\right)$$

其中 $\alpha$ 为过减因子（通常取1~3），$\beta$ 为谱下限因子（0.01~0.1）。增强后的语音通过IFFT重构。

### 步骤1：图像频域去噪

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy import fft, ndimage
from PIL import Image

plt.rcParams['font.sans-serif'] = ['SimHei', 'Arial Unicode MS']
plt.rcParams['axes.unicode_minus'] = False

# 生成测试图像（含纹理和噪声）
np.random.seed(42)
M, N = 256, 256

# 创建合成图像：圆形+矩形+渐变
x = np.arange(M)
y = np.arange(N)
X, Y = np.meshgrid(x, y)
img_clean = np.zeros((M, N))
img_clean[((X - 80) ** 2 + (Y - 80) ** 2) < 30 ** 2] = 200  # 圆
img_clean[140:200, 140:200] = 150  # 矩形
img_clean += 50 * np.sin(2 * np.pi * X / 20) * np.cos(2 * np.pi * Y / 20)  # 纹理
img_clean = img_clean.astype(np.float64)

# 添加高斯噪声
sigma_noise = 30
img_noisy = img_clean + sigma_noise * np.random.randn(M, N)

# 2D FFT
F_clean = fft.fft2(img_clean)
F_noisy = fft.fft2(img_noisy)

# 频谱中心化（低频移到中心）
F_clean_shifted = fft.fftshift(F_clean)
F_noisy_shifted = fft.fftshift(F_noisy)

# 可视化
fig, axes = plt.subplots(2, 2, figsize=(10, 10))
axes[0, 0].imshow(img_clean, cmap='gray')
axes[0, 0].set_title('原始图像')
axes[0, 1].imshow(img_noisy, cmap='gray')
axes[0, 1].set_title(f'含噪图像 (σ={sigma_noise})')
axes[1, 0].imshow(np.log1p(np.abs(F_clean_shifted)), cmap='hot')
axes[1, 0].set_title('原始图像频谱（对数）')
axes[1, 1].imshow(np.log1p(np.abs(F_noisy_shifted)), cmap='hot')
axes[1, 1].set_title('含噪图像频谱（对数）')
for ax in axes.flat:
    ax.axis('off')
plt.tight_layout()
plt.savefig('fft_spectrum_comparison.png', dpi=150)
plt.show()
```

### 步骤2：设计并应用频域滤波器

```python
def create_filter(shape, D0, filter_type='gaussian', order=2):
    """
    创建频域滤波器
    参数:
        shape: (M, N) 图像尺寸
        D0: 截止频率
        filter_type: 'ideal', 'butterworth', 'gaussian'
        order: 巴特沃斯滤波器阶数
    """
    M, N = shape
    u = np.arange(M) - M // 2
    v = np.arange(N) - N // 2
    U, V = np.meshgrid(v, u)  # 注意行列对应
    D = np.sqrt(U ** 2 + V ** 2)

    if filter_type == 'ideal':
        H = np.double(D <= D0)
    elif filter_type == 'butterworth':
        H = 1 / (1 + (D / D0) ** (2 * order))
    elif filter_type == 'gaussian':
        H = np.exp(-D ** 2 / (2 * D0 ** 2))
    return H

# 应用不同滤波器
D0 = 40  # 截止频率
filters = {
    '理想低通': create_filter((M, N), D0, 'ideal'),
    '巴特沃斯低通(n=2)': create_filter((M, N), D0, 'butterworth', order=2),
    '高斯低通': create_filter((M, N), D0, 'gaussian'),
}

fig, axes = plt.subplots(2, 3, figsize=(12, 8))

for idx, (name, H) in enumerate(filters.items()):
    # 频域滤波
    F_filtered = F_noisy_shifted * H
    # 逆变换
    F_back = fft.ifftshift(F_filtered)
    img_filtered = np.real(fft.ifft2(F_back))

    # 计算PSNR
    mse = np.mean((img_clean - img_filtered) ** 2)
    psnr = 10 * np.log10(255 ** 2 / (mse + 1e-10))

    axes[0, idx].imshow(H, cmap='gray')
    axes[0, idx].set_title(f'{name}\n滤波器')
    axes[0, idx].axis('off')

    axes[1, idx].imshow(np.clip(img_filtered, 0, 255), cmap='gray')
    axes[1, idx].set_title(f'{name}\nPSNR={psnr:.1f}dB')
    axes[1, idx].axis('off')

plt.tight_layout()
plt.savefig('frequency_domain_filtering.png', dpi=150)
plt.show()

# 截止频率影响分析
D0_values = [10, 20, 40, 60, 80, 120]
psnr_values = []
for d0 in D0_values:
    H = create_filter((M, N), d0, 'gaussian')
    F_f = F_noisy_shifted * H
    img_f = np.real(fft.ifft2(fft.ifftshift(F_f)))
    mse = np.mean((img_clean - img_f) ** 2)
    psnr_values.append(10 * np.log10(255 ** 2 / (mse + 1e-10)))

plt.figure(figsize=(8, 4))
plt.plot(D0_values, psnr_values, 'bo-', linewidth=2)
plt.xlabel('截止频率 D₀')
plt.ylabel('PSNR (dB)')
plt.title('高斯低通滤波器截止频率对去噪效果的影响')
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig('cutoff_frequency_analysis.png', dpi=150)
plt.show()
```

### 步骤3：谱减法语音增强——生成含噪语音

```python
from scipy.io import wavfile
import os

# 生成合成语音信号
fs = 16000
duration = 3.0
t = np.arange(int(fs * duration)) / fs

# 模拟语音：基频+谐波，带有幅度包络
f0 = 150  # 基频
voiced = np.zeros_like(t)
voiced[0.2 * fs:0.8 * fs] = 1  # 第一段
voiced[1.0 * fs:1.8 * fs] = 1  # 第二段
voiced[2.2 * fs:2.8 * fs] = 1  # 第三段

# 语音信号：多个谐波
speech = voiced * (
    0.6 * np.sin(2 * np.pi * f0 * t) +
    0.3 * np.sin(2 * np.pi * 2 * f0 * t) +
    0.15 * np.sin(2 * np.pi * 3 * f0 * t) +
    0.1 * np.sin(2 * np.pi * 4 * f0 * t)
)

# 添加白噪声
snr_db = 5  # 信噪比 5dB
speech_power = np.mean(speech[voiced > 0] ** 2)
noise_power = speech_power / (10 ** (snr_db / 10))
noise = np.sqrt(noise_power) * np.random.randn(len(t))

noisy_speech = speech + noise
print(f"语音时长: {duration}s, 采样率: {fs}Hz, SNR={snr_db}dB")
```

### 步骤4：谱减法实现

```python
def spectral_subtraction(noisy_signal, fs, frame_size=512, hop_size=256,
                         n_fft=512, alpha=2.0, beta=0.05):
    """
    谱减法语音增强
    参数:
        noisy_signal : 含噪语音
        fs           : 采样率
        frame_size   : 帧长
        hop_size     : 帧移
        n_fft        : FFT点数
        alpha        : 过减因子
        beta         : 谱下限因子
    返回:
        enhanced     : 增强后语音
    """
    # 加窗分帧
    window = np.hanning(frame_size)
    num_frames = (len(noisy_signal) - frame_size) // hop_size + 1

    # STFT
    S_noisy = np.zeros((n_fft // 2 + 1, num_frames), dtype=complex)
    for i in range(num_frames):
        start = i * hop_size
        frame = noisy_signal[start:start + frame_size] * window
        S_noisy[:, i] = np.fft.rfft(frame, n=n_fft)

    # 估计噪声功率谱（取前5帧的平均，假设开头为静音段）
    noise_frames = 5
    noise_psd = np.mean(np.abs(S_noisy[:, :noise_frames]) ** 2, axis=1, keepdims=True)

    # 谱减
    mag_noisy = np.abs(S_noisy)
    phase_noisy = np.angle(S_noisy)

    mag_sq = mag_noisy ** 2
    mag_sub = mag_sq - alpha * noise_psd

    # 半波整流 + 谱下限
    mag_enhanced = np.maximum(np.sqrt(np.maximum(mag_sub, 0)),
                              beta * mag_noisy)

    # 重构
    S_enhanced = mag_enhanced * np.exp(1j * phase_noisy)

    # ISTFT（重叠相加法）
    enhanced = np.zeros(len(noisy_signal))
    window_sum = np.zeros(len(noisy_signal))

    for i in range(num_frames):
        start = i * hop_size
        frame = np.fft.irfft(S_enhanced[:, i], n=n_fft)[:frame_size] * window
        enhanced[start:start + frame_size] += frame
        window_sum[start:start + frame_size] += window ** 2

    # 归一化
    nonzero = window_sum > 1e-10
    enhanced[nonzero] /= window_sum[nonzero]

    return enhanced

# 运行谱减法
enhanced = spectral_subtraction(noisy_speech, fs, alpha=2.0, beta=0.05)

# 计算SNR
def calc_snr(clean, processed):
    noise_res = clean - processed
    return 10 * np.log10(np.sum(clean ** 2) / (np.sum(noise_res ** 2) + 1e-10))

snr_before = calc_snr(speech, noisy_speech)
snr_after = calc_snr(speech, enhanced)
print(f"增强前SNR: {snr_before:.2f} dB")
print(f"增强后SNR: {snr_after:.2f} dB")
print(f"SNR改善: {snr_after - snr_before:.2f} dB")
```
