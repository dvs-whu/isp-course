### 任务二：事件帧累积（Time Surface / Event Frame）

```python
def events_to_frame(events, img_size=(128, 128), num_bins=5):
    """将事件流累积为多帧表示
    参数：
        events: 事件字典
        img_size: 图像尺寸 (H, W)
        num_bins: 时间分帧数
    返回：
        frames: 形状为 (num_bins, H, W) 的累积帧
    """
    t_min, t_max = events['t'][0], events['t'][-1]
    dt = (t_max - t_min) / num_bins
    frames = np.zeros((num_bins, img_size[0], img_size[1]))

    for i in range(len(events['t'])):
        bin_idx = min(int((events['t'][i] - t_min) / dt), num_bins - 1)
        x, y, p = events['x'][i], events['y'][i], events['p'][i]
        frames[bin_idx, y, x] += p  # 正事件+1，负事件-1

    return frames

frames = events_to_frame(events, num_bins=5)

# 可视化各帧
fig, axes = plt.subplots(1, 5, figsize=(20, 4))
for i in range(5):
    axes[i].imshow(frames[i], cmap='RdBu', vmin=-10, vmax=10)
    axes[i].set_title(f'帧 {i+1}')
    axes[i].axis('off')
plt.suptitle('事件帧累积')
plt.tight_layout(); plt.savefig('event_frames.png'); plt.show()
```

## 选项B：语音增强实战

### 任务一：生成含噪语音信号

```python
import numpy as np
import matplotlib.pyplot as plt

# 模拟干净语音：多频率正弦叠加
fs = 16000          # 采样率 16kHz
duration = 2.0      # 2秒
t = np.arange(0, duration, 1/fs)

# 干净语音：基频 + 谐波（模拟语音信号）
clean = 0.6 * np.sin(2 * np.pi * 200 * t)       # 基频 200Hz
clean += 0.3 * np.sin(2 * np.pi * 400 * t)      # 二次谐波
clean += 0.15 * np.sin(2 * np.pi * 600 * t)     # 三次谐波

# 添加高斯白噪声
snr_db = 5  # 信噪比 5dB
noise_power = np.mean(clean**2) / (10 ** (snr_db / 10))
noise = np.sqrt(noise_power) * np.random.randn(len(t))
noisy = clean + noise

print(f"采样率: {fs} Hz, 时长: {duration} s, 样本数: {len(t)}")
print(f"信噪比: {snr_db} dB")
```

### 任务二：STFT 与谱减法去噪

```python
from scipy.signal import stft, istft

# 短时傅里叶变换参数
nperseg  = 512     # 每帧长度
noverlap = 256     # 帧重叠

# STFT
f, t_stft, Zxx_clean = stft(clean, fs=fs, nperseg=nperseg, noverlap=noverlap)
_, _, Zxx_noisy      = stft(noisy, fs=fs, nperseg=nperseg, noverlap=noverlap)

# 幅度谱和相位谱
mag_noisy  = np.abs(Zxx_noisy)
phase_noisy = np.angle(Zxx_noisy)

# 谱减法：估计噪声功率谱（取前5帧平均）
noise_frames = 5
noise_estimate = np.mean(mag_noisy[:, :noise_frames] ** 2, axis=1, keepdims=True)

# 谱减
alpha = 2.0    # 过减因子
beta  = 0.01   # 谱下限
mag_clean_est = np.maximum(mag_noisy**2 - alpha * noise_estimate, beta * np.max(mag_noisy**2))
mag_clean_est = np.sqrt(mag_clean_est)

# 重构信号（用含噪相位）
Zxx_clean_est = mag_clean_est * np.exp(1j * phase_noisy)
_, clean_est = istft(Zxx_clean_est, fs=fs, nperseg=nperseg, noverlap=noverlap)
clean_est = clean_est[:len(t)]  # 截取相同长度

print("谱减法去噪完成")
```

### 任务三：去噪效果评估

```python
def compute_snr(clean, estimate):
    """计算信噪比 (dB)"""
    noise_residue = clean - estimate[:len(clean)]
    snr = 10 * np.log10(np.sum(clean**2) / (np.sum(noise_residue**2) + 1e-10))
    return snr

snr_before = compute_snr(clean, noisy)
snr_after  = compute_snr(clean, clean_est)
print(f"去噪前 SNR: {snr_before:.2f} dB")
print(f"去噪后 SNR: {snr_after:.2f} dB")
print(f"SNR 提升:   {snr_after - snr_before:.2f} dB")
```
